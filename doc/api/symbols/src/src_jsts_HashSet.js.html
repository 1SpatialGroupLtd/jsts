 19 /**
 20  * HashSet
 21  *
 22  * This is a JavaScript implementation of HashSet, similar in concept to those found in Java or C#'s standard libraries.
 23  * It is distributed as part of jshashtable and depends on jshashtable.js. It creates a single constructor function
 24  * called HashSet in the global scope.
 25  *
 26  * Author: Tim Down <tim@timdown.co.uk>
 27  * Version: 2.1
 28  * Build date: 27 March 2010
 29  * Website: http://www.timdown.co.uk/jshashtable/
 30  * @constructor
 31  */
 32 jsts.HashSet = function(hashingFunction, equalityFunction) {
 33   var hashTable = new jsts.Hashtable(hashingFunction, equalityFunction);
 34 
 35   this.add = function(o) {
 36     hashTable.put(o, true);
 37   };
 38 
 39   this.addAll = function(arr) {
 40     var i = arr.length;
 41     while (i--) {
 42       hashTable.put(arr[i], true);
 43     }
 44   };
 45 
 46   this.values = function() {
 47     return hashTable.keys();
 48   };
 49 
 50   this.remove = function(o) {
 51     return hashTable.remove(o) ? o : null;
 52   };
 53 
 54   this.contains = function(o) {
 55     return hashTable.containsKey(o);
 56   };
 57 
 58   this.clear = function() {
 59     hashTable.clear();
 60   };
 61 
 62   this.size = function() {
 63     return hashTable.size();
 64   };
 65 
 66   this.isEmpty = function() {
 67     return hashTable.isEmpty();
 68   };
 69 
 70   this.clone = function() {
 71     var h = new HashSet(hashingFunction, equalityFunction);
 72     h.addAll(hashTable.keys());
 73     return h;
 74   };
 75 
 76   this.intersection = function(hashSet) {
 77     var intersection = new jsts.HashSet(hashingFunction, equalityFunction);
 78     var values = hashSet.values(), i = values.length, val;
 79     while (i--) {
 80       val = values[i];
 81       if (hashTable.containsKey(val)) {
 82         intersection.add(val);
 83       }
 84     }
 85     return intersection;
 86   };
 87 
 88   this.union = function(hashSet) {
 89     var union = this.clone();
 90     var values = hashSet.values(), i = values.length, val;
 91     while (i--) {
 92       val = values[i];
 93       if (!hashTable.containsKey(val)) {
 94         union.add(val);
 95       }
 96     }
 97     return union;
 98   };
 99 
100   this.isSubsetOf = function(hashSet) {
101     var values = hashTable.keys(), i = values.length;
102     while (i--) {
103       if (!hashSet.contains(values[i])) {
104         return false;
105       }
106     }
107     return true;
108   };
109 };
