 19 /**
 20  * jshashtable
 21  *
 22  * jshashtable is a JavaScript implementation of a hash table. It creates a single constructor function called Hashtable
 23  * in the global scope.
 24  *
 25  * Author: Tim Down <tim@timdown.co.uk>
 26  * Version: 2.1
 27  * Build date: 21 March 2010
 28  * Website: http://www.timdown.co.uk/jshashtable
 29  * @constructor
 30  */
 31 jsts.Hashtable = (function() {
 32   var FUNCTION = 'function';
 33 
 34   var arrayRemoveAt = (typeof Array.prototype.splice == FUNCTION) ?
 35       function(arr, idx) {
 36         arr.splice(idx, 1);
 37       } :
 38 
 39       function(arr, idx) {
 40         var itemsAfterDeleted, i, len;
 41         if (idx === arr.length - 1) {
 42           arr.length = idx;
 43         } else {
 44           itemsAfterDeleted = arr.slice(idx + 1);
 45           arr.length = idx;
 46           for (i = 0, len = itemsAfterDeleted.length; i < len; ++i) {
 47             arr[idx + i] = itemsAfterDeleted[i];
 48           }
 49         }
 50       };
 51 
 52   function hashObject(obj) {
 53     var hashCode;
 54     if (typeof obj == 'string') {
 55       return obj;
 56     } else if (typeof obj.hashCode == FUNCTION) {
 57       // Check the hashCode method really has returned a string
 58       hashCode = obj.hashCode();
 59       return (typeof hashCode == 'string') ? hashCode : hashObject(hashCode);
 60     } else if (typeof obj.toString == FUNCTION) {
 61       return obj.toString();
 62     } else {
 63       try {
 64         return String(obj);
 65       } catch (ex) {
 66         // For host objects (such as ActiveObjects in IE) that have no toString() method and throw an error when
 67         // passed to String()
 68         return Object.prototype.toString.call(obj);
 69       }
 70     }
 71   }
 72 
 73   function equals_fixedValueHasEquals(fixedValue, variableValue) {
 74     return fixedValue.equals(variableValue);
 75   }
 76 
 77   function equals_fixedValueNoEquals(fixedValue, variableValue) {
 78     return (typeof variableValue.equals == FUNCTION) ?
 79         variableValue.equals(fixedValue) : (fixedValue === variableValue);
 80   }
 81 
 82   function createKeyValCheck(kvStr) {
 83     return function(kv) {
 84       if (kv === null) {
 85         throw new Error('null is not a valid ' + kvStr);
 86       } else if (typeof kv == 'undefined') {
 87         throw new Error(kvStr + ' must not be undefined');
 88       }
 89     };
 90   }
 91 
 92   var checkKey = createKeyValCheck('key'), checkValue = createKeyValCheck('value');
 93 
 94   /*----------------------------------------------------------------------------------------------------------------*/
 95 
 96   function Bucket(hash, firstKey, firstValue, equalityFunction) {
 97     this[0] = hash;
 98     this.entries = [];
 99     this.addEntry(firstKey, firstValue);
100 
101     if (equalityFunction !== null) {
102       this.getEqualityFunction = function() {
103         return equalityFunction;
104       };
105     }
106   }
107 
108   var EXISTENCE = 0, ENTRY = 1, ENTRY_INDEX_AND_VALUE = 2;
109 
110   function createBucketSearcher(mode) {
111     return function(key) {
112       var i = this.entries.length, entry, equals = this.getEqualityFunction(key);
113       while (i--) {
114         entry = this.entries[i];
115         if (equals(key, entry[0])) {
116           switch (mode) {
117             case EXISTENCE:
118               return true;
119             case ENTRY:
120               return entry;
121             case ENTRY_INDEX_AND_VALUE:
122               return [i, entry[1]];
123           }
124         }
125       }
126       return false;
127     };
128   }
129 
130   function createBucketLister(entryProperty) {
131     return function(aggregatedArr) {
132       var startIndex = aggregatedArr.length;
133       for (var i = 0, len = this.entries.length; i < len; ++i) {
134         aggregatedArr[startIndex + i] = this.entries[i][entryProperty];
135       }
136     };
137   }
138 
139   Bucket.prototype = {
140     getEqualityFunction: function(searchValue) {
141       return (typeof searchValue.equals == FUNCTION) ? equals_fixedValueHasEquals : equals_fixedValueNoEquals;
142     },
143 
144     getEntryForKey: createBucketSearcher(ENTRY),
145 
146     getEntryAndIndexForKey: createBucketSearcher(ENTRY_INDEX_AND_VALUE),
147 
148     removeEntryForKey: function(key) {
149       var result = this.getEntryAndIndexForKey(key);
150       if (result) {
151         arrayRemoveAt(this.entries, result[0]);
152         return result[1];
153       }
154       return null;
155     },
156 
157     addEntry: function(key, value) {
158       this.entries[this.entries.length] = [key, value];
159     },
160 
161     keys: createBucketLister(0),
162 
163     values: createBucketLister(1),
164 
165     getEntries: function(entries) {
166       var startIndex = entries.length;
167       for (var i = 0, len = this.entries.length; i < len; ++i) {
168         // Clone the entry stored in the bucket before adding to array
169         entries[startIndex + i] = this.entries[i].slice(0);
170       }
171     },
172 
173     containsKey: createBucketSearcher(EXISTENCE),
174 
175     containsValue: function(value) {
176       var i = this.entries.length;
177       while (i--) {
178         if (value === this.entries[i][1]) {
179           return true;
180         }
181       }
182       return false;
183     }
184   };
185

188   // Supporting functions for searching hashtable buckets
189 
190   function searchBuckets(buckets, hash) {
191     var i = buckets.length, bucket;
192     while (i--) {
193       bucket = buckets[i];
194       if (hash === bucket[0]) {
195         return i;
196       }
197     }
198     return null;
199   }
200 
201   function getBucketForHash(bucketsByHash, hash) {
202     var bucket = bucketsByHash[hash];
203 
204     // Check that this is a genuine bucket and not something inherited from the bucketsByHash's prototype
205     return (bucket && (bucket instanceof Bucket)) ? bucket : null;
206   }

208   /*----------------------------------------------------------------------------------------------------------------*/
209 
210   function Hashtable(hashingFunctionParam, equalityFunctionParam) {
211     var that = this;
212     var buckets = [];
213     var bucketsByHash = {};
214 
215     var hashingFunction = (typeof hashingFunctionParam == FUNCTION) ? hashingFunctionParam : hashObject;
216     var equalityFunction = (typeof equalityFunctionParam == FUNCTION) ? equalityFunctionParam : null;
217 
218     this.put = function(key, value) {
219       checkKey(key);
220       checkValue(value);
221       var hash = hashingFunction(key), bucket, bucketEntry, oldValue = null;
222 
223       // Check if a bucket exists for the bucket key
224       bucket = getBucketForHash(bucketsByHash, hash);
225       if (bucket) {
226         // Check this bucket to see if it already contains this key
227         bucketEntry = bucket.getEntryForKey(key);
228         if (bucketEntry) {
229           // This bucket entry is the current mapping of key to value, so replace old value and we're done.
230           oldValue = bucketEntry[1];
231           bucketEntry[1] = value;
232         } else {
233           // The bucket does not contain an entry for this key, so add one
234           bucket.addEntry(key, value);
235         }
236       } else {
237         // No bucket exists for the key, so create one and put our key/value mapping in
238         bucket = new Bucket(hash, key, value, equalityFunction);
239         buckets[buckets.length] = bucket;
240         bucketsByHash[hash] = bucket;
241       }
242       return oldValue;
243     };
244 
245     this.get = function(key) {
246       checkKey(key);
247 
248       var hash = hashingFunction(key);
249 
250       // Check if a bucket exists for the bucket key
251       var bucket = getBucketForHash(bucketsByHash, hash);
252       if (bucket) {
253         // Check this bucket to see if it contains this key
254         var bucketEntry = bucket.getEntryForKey(key);
255         if (bucketEntry) {
256           // This bucket entry is the current mapping of key to value, so return the value.
257           return bucketEntry[1];
258         }
259       }
260       return null;
261     };
262 
263     this.containsKey = function(key) {
264       checkKey(key);
265       var bucketKey = hashingFunction(key);
266 
267       // Check if a bucket exists for the bucket key
268       var bucket = getBucketForHash(bucketsByHash, bucketKey);
269 
270       return bucket ? bucket.containsKey(key) : false;
271     };
272 
273     this.containsValue = function(value) {
274       checkValue(value);
275       var i = buckets.length;
276       while (i--) {
277         if (buckets[i].containsValue(value)) {
278           return true;
279         }
280       }
281       return false;
282     };
283 
284     this.clear = function() {
285       buckets.length = 0;
286       bucketsByHash = {};
287     };
288 
289     this.isEmpty = function() {
290       return !buckets.length;
291     };
292 
293     var createBucketAggregator = function(bucketFuncName) {
294       return function() {
295         var aggregated = [], i = buckets.length;
296         while (i--) {
297           buckets[i][bucketFuncName](aggregated);
298         }
299         return aggregated;
300       };
301     };
302 
303     this.keys = createBucketAggregator('keys');
304     this.values = createBucketAggregator('values');
305     this.entries = createBucketAggregator('getEntries');
306 
307     this.remove = function(key) {
308       checkKey(key);
309 
310       var hash = hashingFunction(key), bucketIndex, oldValue = null;
311 
312       // Check if a bucket exists for the bucket key
313       var bucket = getBucketForHash(bucketsByHash, hash);
314 
315       if (bucket) {
316         // Remove entry from this bucket for this key
317         oldValue = bucket.removeEntryForKey(key);
318         if (oldValue !== null) {
319           // Entry was removed, so check if bucket is empty
320           if (!bucket.entries.length) {
321             // Bucket is empty, so remove it from the bucket collections
322             bucketIndex = searchBuckets(buckets, hash);
323             arrayRemoveAt(buckets, bucketIndex);
324             delete bucketsByHash[hash];
325           }
326         }
327       }
328       return oldValue;
329     };
330 
331     this.size = function() {
332       var total = 0, i = buckets.length;
333       while (i--) {
334         total += buckets[i].entries.length;
335       }
336       return total;
337     };
338 
339     this.each = function(callback) {
340       var entries = that.entries(), i = entries.length, entry;
341       while (i--) {
342         entry = entries[i];
343         callback(entry[0], entry[1]);
344       }
345     };
346 
347     this.putAll = function(hashtable, conflictCallback) {
348       var entries = hashtable.entries();
349       var entry, key, value, thisValue, i = entries.length;
350       var hasConflictCallback = (typeof conflictCallback == FUNCTION);
351       while (i--) {
352         entry = entries[i];
353         key = entry[0];
354         value = entry[1];
355 
356         // Check for a conflict. The default behaviour is to overwrite the value for an existing key
357         if (hasConflictCallback && (thisValue = that.get(key))) {
358           value = conflictCallback(key, thisValue, value);
359         }
360         that.put(key, value);
361       }
362     };
363 
364     this.clone = function() {
365       var clone = new jsts.Hashtable(hashingFunctionParam, equalityFunctionParam);
366       clone.putAll(that);
367       return clone;
368     };
369   }
370 
371   return Hashtable;
372 })();
