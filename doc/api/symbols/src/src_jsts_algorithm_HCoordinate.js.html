  8 /**
  9  * Represents a homogeneous coordinate in a 2-D coordinate space. In JTS
 10  * {@link HCoordinate}s are used as a clean way of computing intersections
 11  * between line segments.
 12  *
 13  */
 14 jsts.algorithm.HCoordinate = OpenLayers.Class();
 15 
 16 
 17 /**
 18  * Computes the (approximate) intersection point between two line segments using
 19  * homogeneous coordinates.
 20  * <p>
 21  * Note that this algorithm is not numerically stable; i.e. it can produce
 22  * intersection points which lie outside the envelope of the line segments
 23  * themselves. In order to increase the precision of the calculation input
 24  * points should be normalized before passing them to this routine.
 25  *
 26  * @param {jsts.geom.Coordinate}
 27  *          p1 first coordinate for the first line.
 28  * @param {jsts.geom.Coordinate}
 29  *          p2 second coordinate for the first line.
 30  * @param {jsts.geom.Coordinate}
 31  *          q1 first coordinate for the second line.
 32  * @param {jsts.geom.Coordinate}
 33  *          q2 second coordinate for the second line.
 34  * @return {jsts.geom.Coordinate} The coordinate of the intersection.
 35  */
 36 jsts.algorithm.HCoordinate.intersection = function(p1, p2, q1, q2) {
 37   var px, py, pw, qx, qy, qw, x, y, w, xInt, yInt;
 38 
 39   // unrolled computation
 40   px = p1.y - p2.y;
 41   py = p2.x - p1.x;
 42   pw = p1.x * p2.y - p2.x * p1.y;
 43 
 44   qx = q1.y - q2.y;
 45   qy = q2.x - q1.x;
 46   qw = q1.x * q2.y - q2.x * q1.y;
 47 
 48   x = py * qw - qy * pw;
 49   y = qx * pw - px * qw;
 50   w = px * qy - qx * py;
 51 
 52   xInt = x / w;
 53   yInt = y / w;
 54 
 55   if (!isFinite(xInt) || !isFinite(yInt)) {
 56     throw new jsts.error.NotRepresentableException();
 57   }
 58 
 59   return new jsts.geom.Coordinate(xInt, yInt);
 60 };

 63 /**
 64  * Initializes a new HCoordinate using the OpenLayers inheritance mechanism.
 65  * Will call correct init* function depending on argument.
 66  */
 67 jsts.algorithm.HCoordinate.prototype.initialize = function() {
 68   this.x = 0.0;
 69   this.y = 0.0;
 70   this.w = 1.0;
 71 
 72   if (arguments.length === 1) {
 73     this.initFrom1Coordinate(arguments[0]);
 74   } else if (arguments.length === 2 &&
 75       arguments[0] instanceof jsts.geom.Coordinate) {
 76     this.initFrom2Coordinates(arguments[0], arguments[1]);
 77   } else if (arguments.length === 2 &&
 78       arguments[0] instanceof jsts.algorithm.HCoordinate) {
 79     this.initFrom2HCoordinates(arguments[0], arguments[1]);
 80   } else if (arguments.length === 2) {
 81     this.initFromXY(arguments[0], arguments[1]);
 82   } else if (arguments.length === 3) {
 83     this.initFromXYW(arguments[0], arguments[1], arguments[2]);
 84   } else if (arguments.length === 4) {
 85     this.initFromXYW(arguments[0], arguments[1], arguments[2], arguments[3]);
 86   }
 87 };

 90 /**
 91  * Initializes a new HCoordinate from 1 Coordinate
 92  *
 93  * @param {jsts.geom.Coordinate}
 94  *          p the coordinate.
 95  */
 96 jsts.algorithm.HCoordinate.prototype.initFrom1Coordinate = function(p) {
 97   this.x = p.x;
 98   this.y = p.y;
 99   this.w = 1.0;
100 };

103 /**
104  * Constructs a homogeneous coordinate which is the intersection of the lines
105  * define by the homogenous coordinates represented by two {@link Coordinate}s.
106  *
107  * @param {jsts.geom.Coordinate}
108  *          p1 the first coordinate.
109  * @param {jsts.geom.Coordinate}
110  *          p2 the second coordinate.
111  */
112 jsts.algorithm.HCoordinate.prototype.initFrom2Coordinates = function(p1, p2) {
113   // optimization when it is known that w = 1
114   this.x = p1.y - p2.y;
115   this.y = p2.x - p1.x;
116   this.w = p1.x * p2.y - p2.x * p1.y;
117 };
118 
119 
120 /**
121  * Initializes from 2 HCoordinates
122  *
123  * @param {jsts.algorithm.HCoordinate}
124  *          p1 the first HCoordinate.
125  * @param {jsts.algorithm.HCoordinate}
126  *          p2 the second HCoordinate.
127  */
128 jsts.algorithm.HCoordinate.prototype.initFrom2HCoordinates = function(p1, p2) {
129   this.x = p1.y * p2.w - p2.y * p1.w;
130   this.y = p2.x * p1.w - p1.x * p2.w;
131   this.w = p1.x * p2.y - p2.x * p1.y;
132 };
133 
134 
135 /**
136  * Initializes from x,y,w
137  *
138  * @param {Number}
139  *          x the x-value.
140  * @param {Number}
141  *          y the y-value.
142  * @param {Number}
143  *          w the w-value.
144  */
145 jsts.algorithm.HCoordinate.prototype.initFromXYW = function(x, y, w) {
146   this.x = x;
147   this.y = y;
148   this.w = w;
149 };
150 
151 
152 /**
153  * Initializes from x,y
154  *
155  * @param {Number}
156  *          x the x-value.
157  * @param {Number}
158  *          y the y-value.
159  */
160 jsts.algorithm.HCoordinate.prototype.initFromXY = function(x, y) {
161   this.x = x;
162   this.y = y;
163   this.w = 1.0;
164 };

167 /**
168  * Initializes from 4 Coordinates
169  *
170  * @param {jsts.geom.Coordinate}
171  *          p1 the first coordinate.
172  * @param {jsts.geom.Coordinate}
173  *          p2 the second coordinate.
174  * @param {jsts.geom.Coordinate}
175  *          q1 the first coordinate.
176  * @param {jsts.geom.Coordinate}
177  *          q2 the second coordinate.
178  */
179 jsts.algorithm.HCoordinate.prototype.initFrom4Coordinates = function(p1, p2,
180     q1, q2) {
181   var px, py, pw, qx, qy, qw;
182   // unrolled computation
183   px = p1.y - p2.y;
184   py = p2.x - p1.x;
185   pw = p1.x * p2.y - p2.x * p1.y;
186 
187   qx = q1.y - q2.y;
188   qy = q2.x - q1.x;
189   qw = q1.x * q2.y - q2.x * q1.y;
190 
191   this.x = py * qw - qy * pw;
192   this.y = qx * pw - px * qw;
193   this.w = px * qy - qx * py;
194 };
195 
196 
197 /**
198  * Gets x/w
199  *
200  * @return {Number} x/w.
201  */
202 jsts.algorithm.HCoordinate.prototype.getX = function() {
203   var a = this.x / this.w;
204 
205   if (!isFinite(a)) {
206     throw new jsts.error.NotRepresentableException();
207   }
208   return a;
209 };
210 
211 
212 /**
213  * Gets y/w
214  *
215  * @return {Number} y/w.
216  */
217 jsts.algorithm.HCoordinate.prototype.getY = function() {
218   var a = this.y / this.w;
219 
220   if (!isFinite(a)) {
221     throw new jsts.error.NotRepresentableException();
222   }
223   return a;
224 };
225 
226 
227 /**
228  * Gets a coordinate represented by this HCoordinate
229  *
230  * @return {jst.geom.Coordinate} The coordinate.
231  */
232 jsts.algorithm.HCoordinate.prototype.getCoordinate = function() {
233   var p = new jsts.geom.Coordinate();
234   p.x = this.getX();
235   p.y = this.getY();
236   return p;
237 };
