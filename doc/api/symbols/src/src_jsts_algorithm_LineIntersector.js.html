  9 /**
 10  * A LineIntersector is an algorithm that can both test whether
 11  * two line segments intersect and compute the intersection point
 12  * if they do.
 13  * The intersection point may be computed in a precise or non-precise manner.
 14  * Computing it precisely involves rounding it to an integer.  (This assumes
 15  * that the input coordinates have been made precise by scaling them to
 16  * an integer grid.)
 17  *
 18  * @constructor
 19  */
 20 jsts.algorithm.LineIntersector = function() {
 21   this.inputLines = [[], []];
 22   this.intPt = [];
 23 
 24   this.intPt[0] = new jsts.geom.Coordinate();
 25   this.intPt[1] = new jsts.geom.Coordinate();
 26   // alias the intersection points for ease of reference
 27   this.pa = this.intPt[0];
 28   this.pb = this.intPt[1];
 29   this.result = jsts.algorithm.LineIntersector.NO_INTERSECTION;
 30 };

 57 /**
 58  * Computes the "edge distance" of an intersection point p along a segment. The
 59  * edge distance is a metric of the point along the edge. The metric used is a
 60  * robust and easy to compute metric function. It is <b>not</b> equivalent to
 61  * the usual Euclidean metric. It relies on the fact that either the x or the y
 62  * ordinates of the points in the edge are unique, depending on whether the edge
 63  * is longer in the horizontal or vertical direction.
 64  * <p>
 65  * NOTE: This function may produce incorrect distances for inputs where p is not
 66  * precisely on p1-p2 (E.g. p = (139,9) p1 = (139,10), p2 = (280,1) produces
 67  * distanct 0.0, which is incorrect.
 68  * <p>
 69  * My hypothesis is that the function is safe to use for points which are the
 70  * result of <b>rounding</b> points which lie on the line, but not safe to use
 71  * for <b>truncated</b> points.
 72  *
 73  * @param {Coordinate}
 74  *          p
 75  * @param {Coordinate}
 76  *          p0
 77  * @param {Coordinate}
 78  *          p1
 79  * @return {double}
 80  */
 81 jsts.algorithm.LineIntersector.computeEdgeDistance = function(p, p0, p1) {
 82   var dx = Math.abs(p1.x - p0.x);
 83   var dy = Math.abs(p1.y - p0.y);
 84 
 85   var dist = -1.0; // sentinel value
 86   if (p.equals(p0)) {
 87     dist = 0.0;
 88   } else if (p.equals(p1)) {
 89     if (dx > dy)
 90       dist = dx;
 91     else
 92       dist = dy;
 93   } else {
 94     var pdx = Math.abs(p.x - p0.x);
 95     var pdy = Math.abs(p.y - p0.y);
 96     if (dx > dy)
 97       dist = pdx;
 98     else
 99       dist = pdy;
100     // <FIX>
101     // hack to ensure that non-endpoints always have a non-zero distance
102     if (dist == 0.0 && !p.equals(p0)) {
103       dist = Math.max(pdx, pdy);
104     }
105   }
106   if (dist === 0.0 && !p.equals(p0)) {
107     throw new jsts.error.IllegalArgumentError('Bad distance calculation');
108   }
109   return dist;
110 };

113 /**
114  * This function is non-robust, since it may compute the square of large
115  * numbers. Currently not sure how to improve this.
116  *
117  * @param {Coordinate}
118  *          p
119  * @param {Coordinate}
120  *          p0
121  * @param {Coordinate}
122  *          p1
123  * @return {double}
124  */
125 jsts.algorithm.LineIntersector.nonRobustComputeEdgeDistance = function(p, p1,
126     p2) {
127   var dx = p.x - p1.x;
128   var dy = p.y - p1.y;
129   var dist = Math.sqrt(dx * dx + dy * dy); // dummy value
130   if (!(dist === 0.0 && !p.equals(p1))) {
131     throw new jsts.error.IllegalArgumentError('Invalid distance calculation');
132   }
133   return dist;
134 };

224 /**
225  * Computes the intersection of the lines p1-p2 and p3-p4. This function
226  * computes both the boolean value of the hasIntersection test and the
227  * (approximate) value of the intersection point itself (if there is one).
228  *
229  * @param {Coordinate}
230  *          p1
231  * @param {Coordinate}
232  *          p2
233  * @param {Coordinate}
234  *          p3
235  * @param {Coordinate}
236  *          p4
237  */
238 jsts.algorithm.LineIntersector.prototype.computeIntersection = function(p1, p2,
239     p3, p4) {
240   this.inputLines[0][0] = p1;
241   this.inputLines[0][1] = p2;
242   this.inputLines[1][0] = p3;
243   this.inputLines[1][1] = p4;
244   this.result = this.computeIntersect(p1, p2, p3, p4);
245 };

321 /**
322  * Test whether a point is a intersection point of two line segments. Note that
323  * if the intersection is a line segment, this method only tests for equality
324  * with the endpoints of the intersection segment. It does <b>not</b> return
325  * true if the input point is internal to the intersection segment.
326  *
327  * @param {Coordinate}
328  *          pt
329  * @return {boolean} true if the input point is one of the intersection points.
330  */
331 jsts.algorithm.LineIntersector.prototype.isIntersection = function(pt) {
332   for (var i = 0; i < this.result; i++) {
333     if (this.intPt[i].equals2D(pt)) {
334       return true;
335     }
336   }
337   return false;
338 };

341 /**
342  * Tests whether either intersection point is an interior point of one of the
343  * input segments.
344  *
345  * @return {boolean} <code>true</code> if either intersection point is in the
346  *         interior of one of the input segments.
347  */
348 jsts.algorithm.LineIntersector.prototype.isInteriorIntersection = function() {
349   if (this.isInteriorIntersection(0))
350     return true;
351   if (this.isInteriorIntersection(1))
352     return true;
353   return false;
354 };

357 /**
358  * Tests whether either intersection point is an interior point of the specified
359  * input segment.
360  *
361  * @param {[]} TODO: doc
362  * @return {boolean} <code>true</code> if either intersection point is in the
363  *         interior of the input segment.
364  */
365 jsts.algorithm.LineIntersector.prototype.isInteriorIntersection = function(
366     inputLineIndex) {
367   var i;
368   for (i = 0; i < this.result; i++) {
369     if (!(this.intPt[i].equals2D(this.inputLines[inputLineIndex][0]) || this.intPt[i]
370         .equals2D(this.inputLines[inputLineIndex][1]))) {
371       return true;
372     }
373   }
374   return false;
375 };

378 /**
379  * Tests whether an intersection is proper. <br>
380  * The intersection between two line segments is considered proper if they
381  * intersect in a single point in the interior of both segments (e.g. the
382  * intersection is a single point and is not equal to any of the endpoints).
383  * <p>
384  * The intersection between a point and a line segment is considered proper if
385  * the point lies in the interior of the segment (e.g. is not equal to either of
386  * the endpoints).
387  *
388  * @return {boolean} true if the intersection is proper.
389  */
390 jsts.algorithm.LineIntersector.prototype.isProper = function() {
391   return this.hasIntersection() && this._isProper;
392 };

395 /**
396  * Computes the intIndex'th intersection point in the direction of a specified
397  * input line segment
398  *
399  * @param {int}
400  *          segmentIndex is 0 or 1.
401  * @param {int}
402  *          intIndex is 0 or 1.
403  *
404  * @return {Coordinate} the intIndex'th intersection point in the direction of
405  *         the specified input line segment.
406  */
407 jsts.algorithm.LineIntersector.prototype.getIntersectionAlongSegment = function(
408     segmentIndex, intIndex) {
409   // lazily compute int line array
410   this.computeIntLineIndex();

433 /**
434  * @param {int}
435  *          segmentIndex
436  * @protected
437  */
438 jsts.algorithm.LineIntersector.prototype.computeIntLineIndex = function(
439     segmentIndex) {
440   var dist0 = this.getEdgeDistance(segmentIndex, 0);
441   var dist1 = this.getEdgeDistance(segmentIndex, 1);
442   if (dist0 > dist1) {
443     this.intLineIndex[segmentIndex][0] = 0;
444     this.intLineIndex[segmentIndex][1] = 1;
445   } else {
446     this.intLineIndex[segmentIndex][0] = 1;
447     this.intLineIndex[segmentIndex][1] = 0;
448   }
449 };

452 /**
453  * Computes the "edge distance" of an intersection point along the specified
454  * input line segment.
455  *
456  * @param {int}
457  *          segmentIndex is 0 or 1.
458  * @param {int}
459  *          intIndex is 0 or 1.
460  *
461  * @return {double} the edge distance of the intersection point.
462  */
463 jsts.algorithm.LineIntersector.prototype.getEdgeDistance = function(
464     segmentIndex, intIndex) {
465   var dist = jsts.algorithm.LineIntersector.computeEdgeDistance(this.intPt[intIndex],
466       this.inputLines[segmentIndex][0], this.inputLines[segmentIndex][1]);
467   return dist;
