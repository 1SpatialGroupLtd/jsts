  7 /**
  8  * The base class for all geometric objects.
  9  *
 10  *  <H3>Binary Predicates</H3>
 11  * Because it is not clear at this time
 12  * what semantics for spatial
 13  *  analysis methods involving <code>GeometryCollection</code>s would be useful,
 14  *  <code>GeometryCollection</code>s are not supported as arguments to binary
 15  *  predicates (other than <code>convexHull</code>) or the <code>relate</code>
 16  *  method.
 17  *
 18  *  <H3>Set-Theoretic Methods</H3>
 19  *
 20  *  The spatial analysis methods will
 21  *  return the most specific class possible to represent the result. If the
 22  *  result is homogeneous, a <code>Point</code>, <code>LineString</code>, or
 23  *  <code>Polygon</code> will be returned if the result contains a single
 24  *  element; otherwise, a <code>MultiPoint</code>, <code>MultiLineString</code>,
 25  *  or <code>MultiPolygon</code> will be returned. If the result is
 26  *  heterogeneous a <code>GeometryCollection</code> will be returned. <P>
 27  *
 28  *  Because it is not clear at this time what semantics for set-theoretic
 29  *  methods involving <code>GeometryCollection</code>s would be useful,
 30  * <code>GeometryCollections</code>
 31  *  are not supported as arguments to the set-theoretic methods.
 32  *
 33  *  <H4>Representation of Computed Geometries </H4>
 34  *
 35  *  The SFS states that the result
 36  *  of a set-theoretic method is the "point-set" result of the usual
 37  *  set-theoretic definition of the operation (SFS 3.2.21.1). However, there are
 38  *  sometimes many ways of representing a point set as a <code>Geometry</code>.
 39  *  <P>
 40  *
 41  *  The SFS does not specify an unambiguous representation of a given point set
 42  *  returned from a spatial analysis method. One goal of JTS is to make this
 43  *  specification precise and unambiguous. JTS will use a canonical form for
 44  *  <code>Geometry</code>s returned from spatial analysis methods. The canonical
 45  *  form is a <code>Geometry</code> which is simple and noded:
 46  *  <UL>
 47  *    <LI> Simple means that the Geometry returned will be simple according to
 48  *    the JTS definition of <code>isSimple</code>.
 49  *    <LI> Noded applies only to overlays involving <code>LineString</code>s. It
 50  *    means that all intersection points on <code>LineString</code>s will be
 51  *    present as endpoints of <code>LineString</code>s in the result.
 52  *  </UL>
 53  *  This definition implies that non-simple geometries which are arguments to
 54  *  spatial analysis methods must be subjected to a line-dissolve process to
 55  *  ensure that the results are simple.
 56  *
 57  *  <H4> Constructed Points And The Precision Model </H4>
 58  *
 59  *  The results computed by the set-theoretic methods may contain constructed
 60  *  points which are not present in the input <code>Geometry</code>.
 61  *  These new points arise from intersections between line segments in the
 62  *  edges of the input <code>Geometry</code>s. In the general case it is not
 63  *  possible to represent constructed points exactly. This is due to the fact
 64  *  that the coordinates of an intersection point may contain twice as many bits
 65  *  of precision as the coordinates of the input line segments. In order to
 66  *  represent these constructed points explicitly, JTS must truncate them to fit
 67  *  the <code>PrecisionModel</code>.
 68  *
 69  *  Unfortunately, truncating coordinates moves them slightly. Line segments
 70  *  which would not be coincident in the exact result may become coincident in
 71  *  the truncated representation. This in turn leads to "topology collapses" --
 72  *  situations where a computed element has a lower dimension than it would in
 73  *  the exact result.
 74  *
 75  *  When JTS detects topology collapses during the computation of spatial
 76  *  analysis methods, it will throw an exception. If possible the exception will
 77  *  report the location of the collapse.
 78  *
 79  *  #equals(Object) and #hashCode are not overridden, so that when two
 80  *  topologically equal Geometries are added to HashMaps and HashSets, they
 81  *  remain distinct. This behaviour is desired in many cases.
 82  */

257  * Tests whether this {@link Geometry} is simple. In general, the SFS
258  * specification of simplicity follows the rule:
259  * <UL>
260  * <LI> A Geometry is simple iff the only self-intersections are at boundary

264  * <ul>

278  */
279 jsts.geom.Geometry.prototype.isSimple = function() {
280   this.checkNotGeometryCollection(this);
281   var op = new jsts.operation.IsSimpleOp(this);
282   return op.isSimple();

374 /**
375  * Computes the centroid of this <code>Geometry</code>. The centroid is
376  * equal to the centroid of the set of component Geometries of highest
377  * dimension (since the lower-dimension geometries contribute zero "weight" to
378  * the centroid)
379  *
380  * @return a {@link Point} which is the centroid of this Geometry.
381  */
382 jsts.geom.Geometry.prototype.getCentroid = function() {
383   if (isEmpty()) {
384     return null;
385   }
386   var cent;
387   var centPt = null;
388   var dim = this.getDimension();
389   if (dim === 0) {
390     cent = new CentroidPoint();
391     cent.add(this);
392     centPt = cent.getCentroid();
393   } else if (dim === 1) {
394     cent = new CentroidLine();
395     cent.add(this);
396     centPt = cent.getCentroid();
397   } else {
398     cent = new CentroidArea();
399     cent.add(this);
400     centPt = cent.getCentroid();
401   }
402   return this.createPointFromInternalCoord(centPt, this);
403 
404 };

539 /**
540  * Tests whether this geometry touches the specified geometry.
541  * <p>
542  * The <code>touches</code> predicate has the following equivalent
543  * definitions:
544  * <ul>
545  * <li>The geometries have at least one point in common, but their interiors
546  * do not intersect.
547  * <li>The DE-9IM Intersection Matrix for the two geometries matches
548  * <code>[FT*******]</code> or <code>[F**T*****]</code> or
549  * <code>[F***T****]</code>
550  * </ul>
551  * If both geometries have dimension 0, this predicate returns
552  * <code>false</code>
553  *
554  * @param {Geometry}
555  *          g the <code>Geometry</code> with which to compare this
556  *          <code>Geometry.</code>
557  * @return {boolean} <code>true</code> if the two <code>Geometry</code>s
558  *         touch; Returns <code>false</code> if both <code>Geometry</code>s
559  *         are points.
560  */
561 jsts.geom.Geometry.prototype.touches = function(g) {
562   // short-circuit test
563   if (!this.getEnvelopeInternal().intersects(g.getEnvelopeInternal())) {
564     return false;
565   }
566   return this.relate(g).isTouches(this.getDimension(), g.getDimension());
567 };

570 /**
571  * Tests whether this geometry intersects the specified geometry.
572  * <p>
573  * The <code>intersects</code> predicate has the following equivalent
574  * definitions:
575  * <ul>
576  * <li>The two geometries have at least one point in common
577  * <li>The DE-9IM Intersection Matrix for the two geometries matches
578  * <code>[T********]</code> or <code>[*T*******]</code> or
579  * <code>[***T*****]</code> or <code>[****T****]</code>
580  * <li><code>! g.disjoint(this)</code> (<code>intersects</code> is the
581  * inverse of <code>disjoint</code>)
582  * </ul>
583  *
584  * @param {Geometry}
585  *          g the <code>Geometry</code> with which to compare this
586  *          <code>Geometry.</code>
587  * @return {boolean} <code>true</code> if the two <code>Geometry</code>s
588  *         intersect.
589  *
590  * @see Geometry#disjoint
591  */
592 jsts.geom.Geometry.prototype.intersects = function(g) {
593 
594   // short-circuit envelope test
595   if (!this.getEnvelopeInternal().intersects(g.getEnvelopeInternal())) {
596     return false;
597   }
598 
599   /**
600    * TODO: (MD) Add optimizations: - for P-A case: If P is in env(A), test for
601    * point-in-poly - for A-A case: If env(A1).overlaps(env(A2)) test for
602    * overlaps via point-in-poly first (both ways) Possibly optimize selection
603    * of point to test by finding point of A1 closest to centre of env(A2). (Is
604    * there a test where we shouldn't bother - e.g. if env A is much smaller
605    * than env B, maybe there's no point in testing pt(B) in env(A)?
606    */
607 
608   // optimization for rectangle arguments
609   if (this.isRectangle()) {
610     return RectangleIntersects.intersects(this, g);
611   }
612   if (g.isRectangle()) {
613     return RectangleIntersects.intersects(g, this);
614   }
615   // general case
616   return this.relate(g).isIntersects();
617 };

620 /**
621  * Tests whether this geometry crosses the specified geometry.
622  * <p>
623  * The <code>crosses</code> predicate has the following equivalent
624  * definitions:
625  * <ul>
626  * <li>The geometries have some but not all interior points in common.
627  * <li>The DE-9IM Intersection Matrix for the two geometries matches
628  * <ul>
629  * <li><code>[T*T******]</code> (for P/L, P/A, and L/A situations)
630  * <li><code>[T*****T**]</code> (for L/P, A/P, and A/L situations)
631  * <li><code>[0********]</code> (for L/L situations)
632  * </ul>
633  * </ul>
634  * For any other combination of dimensions this predicate returns
635  * <code>false</code>.
636  * <p>
637  * The SFS defined this predicate only for P/L, P/A, L/L, and L/A situations.
638  * JTS extends the definition to apply to L/P, A/P and A/L situations as well,
639  * in order to make the relation symmetric.
640  *
641  * @param {Geometry}
642  *          g the <code>Geometry</code> with which to compare this
643  *          <code>Geometry.</code>
644  * @return {boolean} <code>true</code> if the two <code>Geometry</code>s
645  *         cross.
646  */
647 jsts.geom.Geometry.prototype.crosses = function(g) {
648   // short-circuit test
649   if (!this.getEnvelopeInternal().intersects(g.getEnvelopeInternal())) {

688  * <p>

793   }

797     return true;
798   }
799   return this.relate(g).isCovers();

1435 /**
1436  * Returns the first non-zero result of <code>compareTo</code> encountered as
1437  * the two <code>Collection</code>s are iterated over. If, by the time one
1438  * of the iterations is complete, no non-zero result has been encountered,
1439  * returns 0 if the other iteration is also complete. If <code>b</code>
1440  * completes before <code>a</code>, a positive number is returned; if a
1441  * before b, a negative number.
1442  *
1443  * @param {Collection}
1444  *          a a <code>Collection</code> of <code>Comparable</code>s.
1445  * @param {Collection}
1446  *          b a <code>Collection</code> of <code>Comparable</code>s.
1447  * @return {int} the first non-zero <code>compareTo</code> result, if any;
1448  *         otherwise, zero.
1449  */
1450 jsts.geom.Geometry.prototype.compare = function(a, b) {
1451   var i = a.iterator();
1452   var j = b.iterator();
1453   while (i.hasNext() && j.hasNext()) {
1454     var aElement = i.next();
1455     var bElement = j.next();
1456     var comparison = aElement.compareTo(bElement);
1457     if (comparison !== 0) {
1458       return comparison;
1459     }
1460   }
1461   if (i.hasNext()) {
1462     return 1;
1463   }
1464   if (j.hasNext()) {
1465     return -1;
1466   }
1467   return 0;
1468 };
