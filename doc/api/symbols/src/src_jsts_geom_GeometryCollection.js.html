 24 /**
 25  * @return {boolean}
 26  */
 27 jsts.geom.GeometryCollection.prototype.isEmpty = function() {
 28   for (var i = 0; i < this.geometries.length; i++) {
 29     if (!this.geometries[i].isEmpty()) {
 30       return false;
 31     }
 32   }
 33   return true;
 34 };

 54 /**
 55  * @param {Geometry} other
 56  * @param {double} tolerance
 57  * @return {boolean}
 58  */
 59 jsts.geom.GeometryCollection.prototype.equalsExact = function(other,  tolerance) {
 60   if (!this.isEquivalentClass(other)) {
 61     return false;
 62   }
 63   if (this.geometries.length !== other.geometries.length) {
 64     return false;
 65   }
 66   for (var i = 0; i < this.geometries.length; i++) {
 67     if (!(this.geometries[i]).equalsExact(other.geometries[i], tolerance)) {
 68       return false;
 69     }
 70   }
 71   return true;
 72 };

 74 jsts.geom.GeometryCollection.prototype.apply = function(filter) {
 75   filter.filter(this);
 76   for (var i = 0; i < this.geometries.length; i++) {
 77     this.geometries[i].apply(filter);
 78   }
 79 };

 82 jsts.geom.GeometryCollection.prototype.getDimension = function() {
 83   var dimension = jsts.geom.Dimension.FALSE;
 84   for (var i = 0; i < this.geometries.length; i++) {
 85     var geometry = this.geometries[i];
 86     // NOTE: special handling since in JTS the parts would be Points.
 87     if (geometry instanceof jsts.geom.Coordinate) {
 88       dimension = Math.max(dimension, 0);
 89     } else {
 90       dimension = Math.max(dimension, geometry.getDimension());
 91     }
 92 
 93   }
 94   return dimension;
 95 };

 98 /**
 99  * @protected
100  */
101 jsts.geom.GeometryCollection.prototype.computeEnvelopeInternal = function() {
102   var envelope = new jsts.geom.Envelope();
103   for (var i = 0; i < this.geometries.length; i++) {
104     var geometry = this.geometries[i];
105     // NOTE: special handling since in JTS the parts would be Points.
106     if (geometry instanceof jsts.geom.Coordinate) {
107       envelope.expandToInclude(new jsts.geom.Envelope(geometry));
108     } else {
109       envelope.expandToInclude(geometry.getEnvelopeInternal());
110     }
111   }
112   return envelope;
113 };

115 OpenLayers.Geometry.Collection = OpenLayers.Class(
116     OpenLayers.Geometry.Collection, jsts.geom.GeometryCollection, {
117       initialize: function(components) {
118         OpenLayers.Geometry.prototype.initialize.apply(this, arguments);
119         this.components = [];
120         if (components != null) {
121           this.addComponents(components);
122         }
123         this.geometries = this.components;
124       }
125     });
