  9 /**
 10  * Supplies a set of utility methods for building Geometry objects from lists
 11  * of Coordinates.
 12  *
 13  * Note that the factory constructor methods do <b>not</b> change the input
 14  * coordinates in any way.
 15  *
 16  * In particular, they are not rounded to the supplied <tt>PrecisionModel</tt>.
 17  * It is assumed that input Coordinates meet the given precision.
 18  *
 19  * Constructs a GeometryFactory that generates Geometries having a floating
 20  * PrecisionModel and a spatial-reference ID of 0.
 21  *
 22  * @constructor
 23  */
 24 jsts.geom.GeometryFactory = function() {
 25 
 26 };
 27 
 28 
 29 /**
 30  * Creates a Point using the given Coordinate; a null Coordinate will create an
 31  * empty Geometry.
 32  *
 33  * @param {Coordinate}
 34  *          coordinate Coordinate to base this Point on.
 35  * @return {Point} A new Point.
 36  */
 37 jsts.geom.GeometryFactory.prototype.createPoint = function(coordinate) {
 38   if (coordinate === null) {
 39     return new jsts.geom.Point();
 40   }
 41   return new jsts.geom.Point(coordinate.x, coordinate.y);
 42 };
 43 
 44 
 45 /**
 46  * Creates a LineString using the given Coordinates; a null or empty array will
 47  * create an empty LineString. Consecutive points must not be equal.
 48  *
 49  * @param {Coordinate[]}
 50  *          coordinates an array without null elements, or an empty array, or
 51  *          null.
 52  * @return {LineString} A new LineString.
 53  */
 54 jsts.geom.GeometryFactory.prototype.createLineString = function(coordinates) {
 55   return new jsts.geom.LineString(coordinates);
 56 };
 57 
 58 
 59 /**
 60  * Creates a LinearRing using the given Coordinates; a null or empty array will
 61  * create an empty LinearRing. The points must form a closed and simple
 62  * linestring. Consecutive points must not be equal.
 63  *
 64  * @param {Coordinate[]}
 65  *          coordinates an array without null elements, or an empty array, or
 66  *          null.
 67  * @return {LinearRing} A new LinearRing.
 68  */
 69 jsts.geom.GeometryFactory.prototype.createLinearRing = function(coordinates) {
 70   return new jsts.geom.LinearRing(coordinates);
 71 };
 72 
 73 
 74 /**
 75  * Constructs a <code>Polygon</code> with the given exterior boundary and
 76  * interior boundaries.
 77  *
 78  * @param {LinearRing}
 79  *          shell the outer boundary of the new <code>Polygon</code>, or
 80  *          <code>null</code> or an empty <code>LinearRing</code> if the
 81  *          empty geometry is to be created.
 82  * @param {LinearRing[]}
 83  *          holes the inner boundaries of the new <code>Polygon</code>, or
 84  *          <code>null</code> or empty <code>LinearRing</code> s if the
 85  *          empty geometry is to be created.
 86  * @return {Polygon} A new Polygon.
 87  */
 88 jsts.geom.GeometryFactory.prototype.createPolygon = function(shell, holes) {
 89   var rings = [shell];
 90 
 91   if (holes !== undefined) {
 92     rings = rings.concat(holes);
 93   }
 94 
 95   return new jsts.geom.Polygon(rings);
 96 };
