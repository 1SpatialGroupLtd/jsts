  9 /**
 10  * Models a Dimensionally Extended Nine-Intersection Model (DE-9IM) matrix. This
 11  * class is used to represent intersection matrices (such as "212FF1FF2")
 12  * capturing the topological relationship between two {@link Geometry}s. It can
 13  * also be represent patterns (such as "T*T******")for matching existing
 14  * matrices.
 15  *
 16  * Methods are provided to:
 17  * <UL>
 18  * <LI> set and query the elements of the matrix in a convenient fashion
 19  * <LI> convert to and from the standard string representation (specified in SFS
 20  * Section 2.1.13.2).
 21  * <LI> test to see if a matrix matches a given pattern string.
 22  * </UL>
 23  * <P>
 24  *
 25  * For a description of the DE-9IM, see the <A
 26  * HREF="http://www.opengis.org/techno/specs.htm">OpenGIS Simple Features
 27  * Specification for SQL</A>.
 28  *
 29  * The entries of the matrix are defined by the constants in the
 30  * {@link Dimension} class. The indices of the matrix represent the topological
 31  * locations that occur in a geometry (Interior, Boundary, Exterior). These are
 32  * provided as constants in the {@link Location} class.
 33  *
 34  * @param {string/IntersectionMatrix}
 35  *          elements
 36  * @constructor
 37  */
 38 jsts.geom.IntersectionMatrix = function(elements) {
 39   var other = elements;
 40 
 41   if (elements === undefined || elements === null) {
 42     this.matrix = [[], [], []];
 43     this.setAll(jsts.geom.Dimension.FALSE);
 44   } else if (typeof elements === 'string') {
 45     this.set(elements);
 46   } else if (other instanceof jsts.geom.IntersectionMatrix) {
 47 
 48     this.matrix[jsts.geom.Location.INTERIOR][jsts.geom.Location.INTERIOR] = other.matrix[jsts.geom.Location.INTERIOR][jsts.geom.Location.INTERIOR];
 49     this.matrix[jsts.geom.Location.INTERIOR][jsts.geom.Location.BOUNDARY] = other.matrix[jsts.geom.Location.INTERIOR][jsts.geom.Location.BOUNDARY];
 50     this.matrix[jsts.geom.Location.INTERIOR][jsts.geom.Location.EXTERIOR] = other.matrix[jsts.geom.Location.INTERIOR][jsts.geom.Location.EXTERIOR];
 51     this.matrix[jsts.geom.Location.BOUNDARY][jsts.geom.Location.INTERIOR] = other.matrix[jsts.geom.Location.BOUNDARY][jsts.geom.Location.INTERIOR];
 52     this.matrix[jsts.geom.Location.BOUNDARY][jsts.geom.Location.BOUNDARY] = other.matrix[jsts.geom.Location.BOUNDARY][jsts.geom.Location.BOUNDARY];
 53     this.matrix[jsts.geom.Location.BOUNDARY][jsts.geom.Location.EXTERIOR] = other.matrix[jsts.geom.Location.BOUNDARY][jsts.geom.Location.EXTERIOR];
 54     this.matrix[jsts.geom.Location.EXTERIOR][jsts.geom.Location.INTERIOR] = other.matrix[jsts.geom.Location.EXTERIOR][jsts.geom.Location.INTERIOR];
 55     this.matrix[jsts.geom.Location.EXTERIOR][jsts.geom.Location.BOUNDARY] = other.matrix[jsts.geom.Location.EXTERIOR][jsts.geom.Location.BOUNDARY];
 56     this.matrix[jsts.geom.Location.EXTERIOR][jsts.geom.Location.EXTERIOR] = other.matrix[jsts.geom.Location.EXTERIOR][jsts.geom.Location.EXTERIOR];
 57   }
 58 };

 70 /**
 71  * Adds one matrix to another. Addition is defined by taking the maximum
 72  * dimension value of each position in the summand matrices.
 73  *
 74  * @param {IntersectionMatrix}
 75  *          im the matrix to add.
 76  */
 77 jsts.geom.IntersectionMatrix.prototype.add = function(im) {
 78   var i, j;
 79   for (i = 0; i < 3; i++) {
 80     for (j = 0; j < 3; j++) {
 81       this.setAtLeast(i, j, im.get(i, j));
 82     }
 83   }
 84 };

 87 /**
 88  * Returns true if the dimension value satisfies the dimension symbol.
 89  *
 90  * @param {int}
 91  *          actualDimensionValue a number that can be stored in the
 92  *          <code>IntersectionMatrix</code> . Possible values are
 93  *          <code>{TRUE, FALSE, DONTCARE, 0, 1, 2}</code>.
 94  * @param {string}
 95  *          requiredDimensionSymbol a character used in the string
 96  *          representation of an <code>IntersectionMatrix</code>. Possible
 97  *          values are <code>{T, F, * , 0, 1, 2}</code>.
 98  * @return {boolean} true if the dimension symbol encompasses the dimension
 99  *         value.
100  */
101 jsts.geom.IntersectionMatrix.matches = function(actualDimensionValue,
102     requiredDimensionSymbol) {
103   if (typeof actualDimensionValue === 'string') {
104     return jsts.geom.IntersectionMatrix.matches2.call(this, arguments);
105   }
106 
107   if (requiredDimensionSymbol === '*') {
108     return true;
109   }
110   if (requiredDimensionSymbol === 'T' &&
111       (actualDimensionValue >= 0 || actualDimensionValue === jsts.geom.Dimension.TRUE)) {
112     return true;
113   }
114   if (requiredDimensionSymbol === 'F' &&
115       actualDimensionValue === jsts.geom.Dimension.FALSE) {
116     return true;
117   }
118   if (requiredDimensionSymbol === '0' &&
119       actualDimensionValue === jsts.geom.Dimension.P) {
120     return true;
121   }
122   if (requiredDimensionSymbol === '1' &&
123       actualDimensionValue === jsts.geom.Dimension.L) {
124     return true;
125   }
126   if (requiredDimensionSymbol === '2' &&
127       actualDimensionValue === jsts.geom.Dimension.A) {
128     return true;
129   }
130   return false;
131 };

134 /**
135  * Returns true if each of the actual dimension symbols satisfies the
136  * corresponding required dimension symbol.
137  *
138  * @param {string}
139  *          actualDimensionSymbols nine dimension symbols to validate. Possible
140  *          values are <code>{T, F, * , 0, 1, 2}</code>.
141  * @param {string}
142  *          requiredDimensionSymbols nine dimension symbols to validate against.
143  *          Possible values are <code>{T, F, * , 0, 1, 2}</code>.
144  * @return {boolean} true if each of the required dimension symbols encompass
145  *         the corresponding actual dimension symbol.
146  */
147 jsts.geom.IntersectionMatrix.matches2 = function(actualDimensionSymbols,
148     requiredDimensionSymbols) {
149   var m = new jsts.geom.IntersectionMatrix(actualDimensionSymbols);
150   return m.matches(requiredDimensionSymbols);
151 };

154 /**
155  * Changes the value of one of this <code>IntersectionMatrix</code>s
156  * elements.
157  *
158  * @param {int}
159  *          row the row of this <code>IntersectionMatrix</code>, indicating
160  *          the interior, boundary or exterior of the first
161  *          <code>Geometry.</code>
162  * @param {int}
163  *          column the column of this <code>IntersectionMatrix</code>,
164  *          indicating the interior, boundary or exterior of the second
165  *          <code>Geometry.</code>
166  * @param {int}
167  *          dimensionValue the new value of the element.
168  */
169 jsts.geom.IntersectionMatrix.prototype.set = function(row, column,
170     dimensionValue) {
171   if (typeof row === 'string') {
172     this.set2(row);
173     return;
174   }
175 
176   this.matrix[row][column] = dimensionValue;
177 };

180 /**
181  * Changes the elements of this <code>IntersectionMatrix</code> to the
182  * dimension symbols in <code>dimensionSymbols</code>.
183  *
184  * @param {String}
185  *          dimensionSymbols nine dimension symbols to which to set this
186  *          <code>IntersectionMatrix</code> s elements. Possible values are
187  *          <code>{T, F, * , 0, 1, 2}.</code>
188  */
189 jsts.geom.IntersectionMatrix.prototype.set2 = function(dimensionSymbols) {
190   for (var i = 0; i < dimensionSymbols.length(); i++) {
191     var row = i / 3;
192     var col = i % 3;
193     this.matrix[row][col] = jsts.geom.Dimension
194         .toDimensionValue(dimensionSymbols.charAt(i));
195   }
196 };

199 /**
200  * Changes the specified element to <code>minimumDimensionValue</code> if the
201  * element is less.
202  *
203  * @param {int}
204  *          row the row of this <code>IntersectionMatrix</code> , indicating
205  *          the interior, boundary or exterior of the first
206  *          <code>Geometry.</code>
207  * @param {int}
208  *          column the column of this <code>IntersectionMatrix</code> ,
209  *          indicating the interior, boundary or exterior of the second
210  *          <code>Geometry.</code>
211  * @param {int}
212  *          minimumDimensionValue the dimension value with which to compare the
213  *          element. The order of dimension values from least to greatest is
214  *          <code>{DONTCARE, TRUE, FALSE, 0, 1, 2}</code>.
215  */
216 jsts.geom.IntersectionMatrix.prototype.setAtLeast = function(row, column,
217     minimumDimensionValue) {
218   if (arguments.length === 1) {
219     this.setAtLeast2(arguments[0]);
220     return;
221   }
222 
223   if (this.matrix[row][column] < minimumDimensionValue) {
224     this.matrix[row][column] = minimumDimensionValue;
225   }
226 };

229 /**
230  * If row >= 0 and column >= 0, changes the specified element to
231  * <code>minimumDimensionValue</code> if the element is less. Does nothing if
232  * row <0 or column < 0.
233  *
234  * @param {int}
235  *          row the row of this <code>IntersectionMatrix</code> , indicating
236  *          the interior, boundary or exterior of the first
237  *          <code>Geometry.</code>
238  * @param {int}
239  *          column the column of this <code>IntersectionMatrix</code> ,
240  *          indicating the interior, boundary or exterior of the second
241  *          <code>Geometry.</code>
242  * @param {int}
243  *          minimumDimensionValue the dimension value with which to compare the
244  *          element. The order of dimension values from least to greatest is
245  *          <code>{DONTCARE, TRUE, FALSE, 0, 1, 2}</code>.
246  */
247 jsts.geom.IntersectionMatrix.prototype.setAtLeastIfValid = function(row,
248     column, minimumDimensionValue) {
249   if (row >= 0 && column >= 0) {
250     this.setAtLeast(row, column, minimumDimensionValue);
251   }
252 };

255 /**
256  * For each element in this <code>IntersectionMatrix</code>, changes the
257  * element to the corresponding minimum dimension symbol if the element is less.
258  *
259  * @param {string}
260  *          minimumDimensionSymbols nine dimension symbols with which to compare
261  *          the elements of this <code>IntersectionMatrix</code>. The order
262  *          of dimension values from least to greatest is
263  *          <code>{DONTCARE, TRUE, FALSE, 0, 1, 2}</code> .
264  */
265 jsts.geom.IntersectionMatrix.prototype.setAtLeast2 = function(
266     minimumDimensionSymbols) {
267   var i;
268   for (i = 0; i < minimumDimensionSymbols.length; i++) {
269     var row = parseInt(i / 3);
270     var col = parseInt(i % 3);
271     this.setAtLeast(row, col, jsts.geom.Dimension
272         .toDimensionValue(minimumDimensionSymbols.charAt(i)));
273   }
274 };

316 /**
317  * Returns <code>true</code> if this <code>IntersectionMatrix</code> is
318  * FF*FF****.
319  *
320  * @return {boolean} <code>true</code> if the two <code>Geometry</code>s
321  *         related by this <code>IntersectionMatrix</code> are disjoint.
322  */
323 jsts.geom.IntersectionMatrix.prototype.isDisjoint = function() {
324   return this.matrix[jsts.geom.Location.INTERIOR][jsts.geom.Location.INTERIOR] === jsts.geom.Dimension.FALSE &&
325       this.matrix[jsts.geom.Location.INTERIOR][jsts.geom.Location.BOUNDARY] === jsts.geom.Dimension.FALSE &&
326       this.matrix[jsts.geom.Location.BOUNDARY][jsts.geom.Location.INTERIOR] === jsts.geom.Dimension.FALSE &&
327       this.matrix[jsts.geom.Location.BOUNDARY][jsts.geom.Location.BOUNDARY] === jsts.geom.Dimension.FALSE;
328 };
329 
330 
331 /**
332  * Returns <code>true</code> if <code>isDisjoint</code> returns false.
333  *
334  * @return {boolean} <code>true</code> if the two <code>Geometry</code>s
335  *         related by this <code>IntersectionMatrix</code> intersect.
336  */
337 jsts.geom.IntersectionMatrix.prototype.isIntersects = function() {
338   return !this.isDisjoint();

342 /**
343  * Returns <code>true</code> if this <code>IntersectionMatrix</code> is
344  * FT*******, F**T***** or F***T****.
345  *
346  * @param {int}
347  *          dimensionOfGeometryA the dimension of the first
348  *          <code>Geometry.</code>
349  * @param {int}
350  *          dimensionOfGeometryB the dimension of the second
351  *          <code>Geometry.</code>
352  * @return {boolean} <code>true</code> if the two <code>Geometry</code> s
353  *         related by this <code>IntersectionMatrix</code> touch; Returns
354  *         false if both <code>Geometry</code>s are points.
355  */
356 jsts.geom.IntersectionMatrix.prototype.isTouches = function(
357     dimensionOfGeometryA, dimensionOfGeometryB) {
358   if (dimensionOfGeometryA > dimensionOfGeometryB) {
359     // no need to get transpose because pattern matrix is symmetrical
360     return this.isTouches(dimensionOfGeometryB, dimensionOfGeometryA);
361   }
362   if ((dimensionOfGeometryA == jsts.geom.Dimension.A && dimensionOfGeometryB == jsts.geom.Dimension.A) ||
363       (dimensionOfGeometryA == jsts.geom.Dimension.L && dimensionOfGeometryB == jsts.geom.Dimension.L) ||
364       (dimensionOfGeometryA == jsts.geom.Dimension.L && dimensionOfGeometryB == jsts.geom.Dimension.A) ||
365       (dimensionOfGeometryA == jsts.geom.Dimension.P && dimensionOfGeometryB == jsts.geom.Dimension.A) ||
366       (dimensionOfGeometryA == jsts.geom.Dimension.P && dimensionOfGeometryB == jsts.geom.Dimension.L)) {
367     return this.matrix[jsts.geom.Location.INTERIOR][jsts.geom.Location.INTERIOR] === jsts.geom.Dimension.FALSE &&
368         (jsts.geom.IntersectionMatrix
369             .matches(
370                 this.matrix[jsts.geom.Location.INTERIOR][jsts.geom.Location.BOUNDARY],
371                 'T') ||
372             jsts.geom.IntersectionMatrix
373                 .matches(
374                     this.matrix[jsts.geom.Location.BOUNDARY][jsts.geom.Location.INTERIOR],
375                     'T') || jsts.geom.IntersectionMatrix
376             .matches(
377                 this.matrix[jsts.geom.Location.BOUNDARY][jsts.geom.Location.BOUNDARY],
378                 'T'));
379   }
380   return false;
381 };

384 /**
385  * Tests whether this geometry crosses the specified geometry.
386  * <p>
387  * The <code>crosses</code> predicate has the following equivalent
388  * definitions:
389  * <ul>
390  * <li>The geometries have some but not all interior points in common.
391  * <li>The DE-9IM Intersection Matrix for the two geometries is
392  * <ul>
393  * <li>T*T****** (for P/L, P/A, and L/A situations)
394  * <li>T*****T** (for L/P, L/A, and A/L situations)
395  * <li>0******** (for L/L situations)
396  * </ul>
397  * </ul>
398  * For any other combination of dimensions this predicate returns
399  * <code>false</code>.
400  * <p>
401  * The SFS defined this predicate only for P/L, P/A, L/L, and L/A situations.
402  * JTS extends the definition to apply to L/P, A/P and A/L situations as well.
403  * This makes the relation symmetric.
404  *
405  * @param {int}
406  *          dimensionOfGeometryA the dimension of the first
407  *          <code>Geometry.</code>
408  * @param {int}
409  *          dimensionOfGeometryB the dimension of the second
410  *          <code>Geometry.</code>
411  * @return {boolean} <code>true</code> if the two <code>Geometry</code>s
412  *         related by this <code>IntersectionMatrix</code> cross.
413  */
414 jsts.geom.IntersectionMatrix.prototype.isCrosses = function(
415     dimensionOfGeometryA, dimensionOfGeometryB) {
416   if ((dimensionOfGeometryA == jsts.geom.Dimension.P && dimensionOfGeometryB == jsts.geom.Dimension.L) ||
417       (dimensionOfGeometryA == jsts.geom.Dimension.P && dimensionOfGeometryB == jsts.geom.Dimension.A) ||
418       (dimensionOfGeometryA == jsts.geom.Dimension.L && dimensionOfGeometryB == jsts.geom.Dimension.A)) {
419     return jsts.geom.IntersectionMatrix.matches(
420         this.matrix[jsts.geom.Location.INTERIOR][jsts.geom.Location.INTERIOR],
421         'T') &&
422         jsts.geom.IntersectionMatrix
423             .matches(
424                 this.matrix[jsts.geom.Location.INTERIOR][jsts.geom.Location.EXTERIOR],
425                 'T');
426   }
427   if ((dimensionOfGeometryA == jsts.geom.Dimension.L && dimensionOfGeometryB == jsts.geom.Dimension.P) ||
428       (dimensionOfGeometryA == jsts.geom.Dimension.A && dimensionOfGeometryB == jsts.geom.Dimension.P) ||
429       (dimensionOfGeometryA == jsts.geom.Dimension.A && dimensionOfGeometryB == jsts.geom.Dimension.L)) {
430     return jsts.geom.IntersectionMatrix.matches(
431         matrix[jsts.geom.Location.INTERIOR][jsts.geom.Location.INTERIOR], 'T') &&
432         jsts.geom.IntersectionMatrix
433             .matches(
434                 this.matrix[jsts.geom.Location.EXTERIOR][jsts.geom.Location.INTERIOR],
435                 'T');
436   }
437   if (dimensionOfGeometryA === jsts.geom.Dimension.L &&
438       dimensionOfGeometryB === jsts.geom.Dimension.L) {
439     return this.matrix[jsts.geom.Location.INTERIOR][jsts.geom.Location.INTERIOR] === 0;
440   }
441   return false;
442 };

537 /**
538  * Returns <code>true</code> if this <code>IntersectionMatrix</code> is
539  * T*F**FFF*.
540  *
541  * @param {int}
542  *          dimensionOfGeometryA the dimension of the first
543  *          <code>Geometry.</code>
544  * @param {int}
545  *          dimensionOfGeometryB the dimension of the second
546  *          <code>Geometry.</code>
547  * @return {boolean} <code>true</code> if the two <code>Geometry</code> s
548  *         related by this <code>IntersectionMatrix</code> are equal; the
549  *         <code>Geometry</code>s must have the same dimension for this
550  *         function to return <code>true.</code>
551  */
552 jsts.geom.IntersectionMatrix.prototype.isEquals = function(
553     dimensionOfGeometryA, dimensionOfGeometryB) {
554   if (dimensionOfGeometryA !== dimensionOfGeometryB) {
555     return false;
556   }
557   return jsts.geom.IntersectionMatrix.matches(
558       this.matrix[jsts.geom.Location.INTERIOR][jsts.geom.Location.INTERIOR],
559       'T') &&
560       this.matrix[jsts.geom.Location.EXTERIOR][jsts.geom.Location.INTERIOR] === jsts.geom.Dimension.FALSE &&
561       this.matrix[jsts.geom.Location.INTERIOR][jsts.geom.Location.EXTERIOR] === jsts.geom.Dimension.FALSE &&
562       this.matrix[jsts.geom.Location.EXTERIOR][jsts.geom.Location.BOUNDARY] === jsts.geom.Dimension.FALSE &&
563       this.matrix[jsts.geom.Location.BOUNDARY][jsts.geom.Location.EXTERIOR] === jsts.geom.Dimension.FALSE;
564 };

567 /**
568  * Returns <code>true</code> if this <code>IntersectionMatrix</code> is
569  * <UL>
570  * <LI> T*T***T** (for two points or two surfaces)
571  * <LI> 1*T***T** (for two curves)
572  * </UL>.
573  *
574  * @param {int}
575  *          dimensionOfGeometryA the dimension of the first
576  *          <code>Geometry.</code>
577  * @param {int}
578  *          dimensionOfGeometryB the dimension of the second
579  *          <code>Geometry.</code>
580  * @return {boolean} <code>true</code> if the two <code>Geometry</code>s
581  *         related by this <code>IntersectionMatrix</code> overlap. For this
582  *         function to return <code>true</code>, the <code>Geometry</code>s
583  *         must be two points, two curves or two surfaces.
584  */
585 jsts.geom.IntersectionMatrix.prototype.isOverlaps = function(
586     dimensionOfGeometryA, dimensionOfGeometryB) {
587   if ((dimensionOfGeometryA == jsts.geom.Dimension.P && dimensionOfGeometryB === jsts.geom.Dimension.P) ||
588       (dimensionOfGeometryA == jsts.geom.Dimension.A && dimensionOfGeometryB === jsts.geom.Dimension.A)) {
589     return jsts.geom.IntersectionMatrix.matches(
590         this.matrix[jsts.geom.Location.INTERIOR][jsts.geom.Location.INTERIOR],
591         'T') &&
592         jsts.geom.IntersectionMatrix
593             .matches(
594                 this.matrix[jsts.geom.Location.INTERIOR][jsts.geom.Location.EXTERIOR],
595                 'T') &&
596         jsts.geom.IntersectionMatrix
597             .matches(
598                 this.matrix[jsts.geom.Location.EXTERIOR][jsts.geom.Location.INTERIOR],
599                 'T');
600   }
601   if (dimensionOfGeometryA === jsts.geom.Dimension.L &&
602       dimensionOfGeometryB === jsts.geom.Dimension.L) {
603     return this.matrix[jsts.geom.Location.INTERIOR][jsts.geom.Location.INTERIOR] == 1 &&
604         jsts.geom.IntersectionMatrix
605             .matches(
606                 this.matrix[jsts.geom.Location.INTERIOR][jsts.geom.Location.EXTERIOR],
607                 'T') &&
608         jsts.geom.IntersectionMatrix
609             .matches(
610                 this.matrix[jsts.geom.Location.EXTERIOR][jsts.geom.Location.INTERIOR],
611                 'T');
612   }
613   return false;
614 };

617 /**
618  * Returns whether the elements of this <code>IntersectionMatrix</code>
619  * satisfies the required dimension symbols.
620  *
621  * @param {string}
622  *          requiredDimensionSymbols nine dimension symbols with which to
623  *          compare the elements of this <code>IntersectionMatrix</code>.
624  *          Possible values are <code>{T, F, * , 0, 1, 2}</code>.
625  * @return {boolean} <code>true</code> if this <code>IntersectionMatrix</code>
626  *         matches the required dimension symbols.
627  */
628 jsts.geom.IntersectionMatrix.prototype.matches = function(
629     requiredDimensionSymbols) {
630   if (requiredDimensionSymbols.length != 9) {
631     throw new jsts.error.IllegalArgumentException('Should be length 9: ' +
632         requiredDimensionSymbols);
633   }
634   for (var ai = 0; ai < 3; ai++) {
635     for (var bi = 0; bi < 3; bi++) {
636       if (!jsts.geom.IntersectionMatrix.matches(this.matrix[ai][bi],
637           requiredDimensionSymbols.charAt(3 * ai + bi))) {
638         return false;
639       }
640     }
641   }
642   return true;
643 };

646 /**
647  * Transposes this IntersectionMatrix.
648  *
649  * @return {IntersectionMatrix} this <code>IntersectionMatrix</code> as a
650  *         convenience.
651  */
652 jsts.geom.IntersectionMatrix.prototype.transpose = function() {
653   var temp = matrix[1][0];
654   this.matrix[1][0] = this.matrix[0][1];
655   this.matrix[0][1] = temp;
656   temp = this.matrix[2][0];
657   this.matrix[2][0] = this.matrix[0][2];
658   this.matrix[0][2] = temp;
659   temp = this.matrix[2][1];
660   this.matrix[2][1] = this.matrix[1][2];
661   this.matrix[1][2] = temp;
662   return this;
663 };

666 /**
667  * Returns a nine-character <code>String</code> representation of this
668  * <code>IntersectionMatrix</code> .
669  *
670  * @return {string} the nine dimension symbols of this
671  *         <code>IntersectionMatrix</code> in row-major order.
672  */
673 jsts.geom.IntersectionMatrix.prototype.toString = function() {
674   var ai, bi, buf = '';
675   for (ai = 0; ai < 3; ai++) {
676     for (bi = 0; bi < 3; bi++) {
677       buf += jsts.geom.Dimension.toDimensionSymbol(this.matrix[ai][bi]);
678     }
679   }
680   return buf;
