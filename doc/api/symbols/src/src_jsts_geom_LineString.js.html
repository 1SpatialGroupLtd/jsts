 42 /**
 43  * @return {jsts.geom.Coordinate} The first coordinate of this LineString or
 44  *         null if empty.
 45  */
 46 jsts.geom.LineString.prototype.getCoordinate = function() {
 47   if (this.isEmpty()) {
 48     return null;
 49   }
 50   return this.getCoordinateN(0);
 51 };

 62 /**
 63  * @return {int} dimension of the boundary of this LineString.
 64  */
 65 jsts.geom.LineString.prototype.getBoundaryDimension = function() {
 66   if (this.isClosed()) {
 67     return Dimension.FALSE;
 68   }
 69   return 0;
 70 };

 73 /**
 74  * @return {Boolean} true if empty.
 75  */
 76 jsts.geom.LineString.prototype.isEmpty = function() {
 77   return this.components.length === 0;
 78 };

 81 /**
 82  * @return {Boolean} true if LineString is Closed.
 83  */
 84 jsts.geom.LineString.prototype.isClosed = function() {
 85   if (this.isEmpty()) {
 86     return false;
 87   }
 88   return this.getCoordinateN(0).equals2D(
 89       this.getCoordinateN(this.components.length - 1));
 90 };
 91 
 92 
 93 /**
 94  * @return {Boolean} true if LineString is a Ring.
 95  */
 96 jsts.geom.LineString.prototype.isRing = function() {
 97   return this.isClosed() && this.isSimple();
 98 };

121 jsts.geom.LineString.prototype.computeEnvelopeInternal = function() {
122   if (this.isEmpty()) {
123     return new jsts.geom.Envelope();
124   }
125 
126   var env = new jsts.geom.Envelope();
127   for (var i = 0; i < this.components.length; i++) {
128     var point = this.components[i];
129     env.expandToInclude(point);
130   }
131 
132   return env;
133 };

136 /**
137  * @param {Geometry}
138  *          other Geometry to compare this LineString to.
139  * @param {double}
140  *          tolerance Tolerance.
141  * @return {Boolean} true if equal.
142  */
143 jsts.geom.LineString.prototype.equalsExact = function(other, tolerance) {
144   var i;
145 
146   if (!this.isEquivalentClass(other)) {
147     return false;
148   }
149   if (this.components.length !== other.components.length) {
150     return false;
151   }
152   for (i = 0; i < this.components.length; i++) {
153     if (!jsts.geom.Geometry.prototype.equal(this.components[i], other.components[i],
154         tolerance)) {
155       return false;
156     }
157   }
158   return true;
159 };

162 /**
163  * Creates and returns a full copy of this {@link LineString} object. (including
164  * all coordinates contained by it).
165  *
166  * @return {jsts.geom.LineString} a clone of this instance.
167  */
168 jsts.geom.LineString.prototype.clone = function() {
169   var key, coordinate;
170 
171   var points = [];
172   for (key in this.components) {
173     if (this.components.hasOwnProperty(key)) {
174       coordinate = this.components[key];
175       points.push(coordinate.clone());
176     }
177   }
178 
179   var clone = new jsts.geom.LineString(points);
180 
181   return clone;
182 };
