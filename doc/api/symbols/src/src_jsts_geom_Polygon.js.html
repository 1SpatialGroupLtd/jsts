  7 /**
  8  * Represents a linear polygon, which may include holes.
  9  * The shell and holes of the polygon are represented by {@link LinearRing}s.
 10  * In a valid polygon, holes may touch the shell or other holes at a single
 11  * point. However, no sequence of touching holes may split the polygon into
 12  * two pieces. The orientation of the rings in the polygon does not matter.
 13  *
 14  * The shell and holes must conform to the assertions specified in the <A
 15  * HREF="http://www.opengis.org/techno/specs.htm">OpenGIS Simple Features
 16  * Specification for SQL</A>.
 17  */
 18

 38 /**
 39  * @return {boolean}
 40  */
 41 jsts.geom.Polygon.prototype.isEmpty = function() {
 42   for (var i = 0; i < this.geometries.length; i++) {
 43     if (!this.geometries[i].isEmpty()) {
 44       return false;
 45     }
 46   }
 47   return true;
 48 };

 65 /**
 66  * Computes the boundary of this geometry
 67  *
 68  * @return {Geometry} a lineal geometry (which may be empty).
 69  * @see Geometry#getBoundary
 70  */
 71 jsts.geom.Polygon.prototype.getBoundary = function() {
 72   if (this.isEmpty()) {
 73     return this.getFactory().createMultiLineString(null);
 74   }
 75   var rings = [];
 76   var shell = this.components[0];
 77   rings[0] = shell;
 78   var holes = this.components.slice(1);
 79   for (var i = 0; i < holes.length; i++) {
 80     rings[i + 1] = holes[i];
 81   }
 82   // create LineString or MultiLineString as appropriate
 83   if (rings.length <= 1)
 84     return this.getFactory().createLinearRing(rings[0].getCoordinateSequence());
 85   return this.getFactory().createMultiLineString(rings);
 86 };

103 /**
104  * @param {Geometry} other
105  * @param {double} tolerance
106  * @return {boolean}
107  */
108 jsts.geom.Polygon.prototype.equalsExact = function(other, tolerance) {
109   if (!this.isEquivalentClass(other)) {
110     return false;
111   }
112   if (this.isEmpty() && other.isEmpty()) {
113     return true;
114   }
115 
116   var holes = this.components.slice(1);
117   var otherPolygon = other;
118   var thisShell = this.components[0];
119   var otherPolygonShell = otherPolygon.components[0];
120   var otherPolygonHoles = otherPolygon.components.slice(1);
121   if (!thisShell.equalsExact(otherPolygonShell, tolerance)) {
122     return false;
123   }
124   if (holes.length !== otherPolygonHoles.length) {
125     return false;
126   }
127   if (holes.length !== otherPolygonHoles.length) {
128     return false;
129   }
130   for (var i = 0; i < holes.length; i++) {
131     if (!(holes[i]).equalsExact(otherPolygonHoles[i], tolerance)) {
132       return false;
133     }
134   }
135   return true;
136 };

138 jsts.geom.Polygon.prototype.apply = function(filter) {
139   filter.filter(this);
140   if (filter instanceof jsts.geom.GeometryComponentFilter) {
141     var shell = this.components[0];
142     shell.apply(filter);
143     var holes = this.components.slice(1);
144     for (var i = 0; i < holes.length; i++) {
145       holes[i].apply(filter);
146     }
147   }
148 };
