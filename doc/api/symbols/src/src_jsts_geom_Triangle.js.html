 16 /**
 17  * Tests whether a triangle is acute.
 18  * A triangle is acute iff all interior angles are acute.
 19  * This is a strict test - right triangles will return <tt>false</tt>
 20  * A triangle which is not acute is either right or obtuse.
 21  * <p>
 22  * Note: this implementation is not robust for angles very close to 90 degrees.
 23  *
 24  * @param {jsts.geom.Coordinate}
 25  *        a a vertex of the triangle.
 26  * @param {jsts.geom.Coordinate}
 27  *        b a vertex of the triangle.
 28  * @param {jsts.geom.Coordinate}
 29  *        c a vertex of the triangle.
 30  * @return {Boolean}
 31  *         true if the triangle is acute.
 32  */
 33 jsts.geom.Triangle.isAcute = function(a, b, c) {
 34   if (! jsts.algorithm.Angle.isAcute(a, b, c)) {
 35     return false;
 36   }
 37   if (! jsts.algorithm.Angle.isAcute(b, c, a)) {
 38     return false;
 39   }
 40   if (! jsts.algorithm.Angle.isAcute(c, a, b)) {
 41     return false;
 42   }
 43   return true;
 44 };

 47 /**
 48  * Computes the line which is the perpendicular bisector of the
 49  * line segment a-b.
 50  *
 51  * @param {jsts.geom.Coordinate}
 52  *        a a point.
 53  * @param {jsts.geom.Coordinate}
 54  *        b another point.
 55  * @return {jsts.algorithm.HCoordinate}
 56  *         the perpendicular bisector, as an HCoordinate.
 57  */
 58 jsts.geom.Triangle.perpendicularBisector = function(a, b) {
 59   //returns the perpendicular bisector of the line segment ab
 60   var dx, dy, l1, l2;
 61 
 62   dx = b.x - a.x;
 63   dy = b.y - a.y;
 64 
 65   l1 = new jsts.algorithm.HCoordinate(a.x + dx / 2.0,
 66       a.y + dy / 2.0, 1.0);
 67   l2 = new jsts.algorithm.HCoordinate(a.x - dy + dx / 2.0,
 68       a.y + dx + dy / 2.0, 1.0);
 69   return new jsts.algorithm.HCoordinate(l1, l2);
 70 };

 73 /**
 74  * Computes the circumcentre of a triangle.
 75  * The circumcentre is the centre of the circumcircle,
 76  * the smallest circle which encloses the triangle.
 77  * It is also the common intersection point of the
 78  * perpendicular bisectors of the sides of the triangle,
 79  * and is the only point which has equal distance to all three
 80  * vertices of the triangle.
 81  * <p>
 82  * This method uses an algorithm due to J.R.Shewchuk which
 83  * uses normalization to the origin
 84  * to improve the accuracy of computation.
 85  * (See <i>Lecture Notes on Geometric Robustness</i>,
 86  * Jonathan Richard Shewchuk, 1999).
 87  *
 88  * @param {jsts.geom.Coordinate}
 89  *        a a vertx of the triangle.
 90  * @param {jsts.geom.Coordinate}
 91  *        b a vertx of the triangle.
 92  * @param {jsts.geom.Coordinate}
 93  *        c a vertx of the triangle.
 94  * @return {jsts.geom.Coordinate}
 95  *         the circumcentre of the triangle.
 96  */
 97 jsts.geom.Triangle.circumcentre = function(a, b, c) {
 98   var cx, cy, ax, ay, bx, by, denom, numx, numy, ccx, ccy;
 99 
100   cx = c.x;
101   cy = c.y;
102   ax = a.x - cx;
103   ay = a.y - cy;
104   bx = b.x - cx;
105   by = b.y - cy;
106 
107   denom = 2 * jsts.geom.Triangle.det(ax, ay, bx, by);
108   numx = jsts.geom.Triangle.det(ay, ax * ax + ay * ay, by,
109       bx * bx + by * by);
110   numy = jsts.geom.Triangle.det(ax, ax * ax + ay * ay, bx,
111       bx * bx + by * by);
112 
113   ccx = cx - numx / denom;
114   ccy = cy + numy / denom;
115 
116   return new jsts.geom.Coordinate(ccx, ccy);
117 };

120 /**
121  * Computes the determinant of a 2x2 matrix.
122  * Uses standard double-precision arithmetic,
123  * so is susceptible to round-off error.
124  *
125  * @param {Number}
126  *        m00 the [0,0] entry of the matrix.
127  * @param {Number}
128  *        m01 the [0,1] entry of the matrix.
129  * @param {Number}
130  *        m10 the [1,0] entry of the matrix.
131  * @param {Number}
132  *        m11 the [1,1] entry of the matrix.
133  * @return {Number}
134  *         the determinant.
135  */
136 jsts.geom.Triangle.det = function(m00, m01, m10, m11) {
137   return m00 * m11 - m01 * m10;
138 };

141 /**
142  * Computes the incentre of a triangle.
143  * The <i>inCentre</i> of a triangle is the point which is equidistant
144  * from the sides of the triangle.
145  * It is also the point at which the bisectors
146  * of the triangle's angles meet.
147  * It is the centre of the triangle's <i>incircle</i>,
148  * which is the unique circle that is tangent to each of the triangle's
149  * three sides.
150  *
151  * @param {jsts.geom.Coordinate}
152  *        a a vertx of the triangle.
153  * @param {jsts.geom.Coordinate}
154  *        b a vertx of the triangle.
155  * @param {jsts.geom.Coordinate}
156  *        c a vertx of the triangle.
157  * @return {jsts.geom.Coordinate}
158  *         the point which is the incentre of the triangle.
159  */
160 jsts.geom.Triangle.inCentre = function(a, b, c) {
161   var len0, len1, len2, circum, inCentreX, inCentreY;
162 
163   //the lengths of the sides, labelled by their opposite vertex
164   len0 = b.distance(c);
165   len1 = a.distance(c);
166   len2 = a.distance(b);
167   circum = len0 + len1 + len2;
168 
169   inCentreX = (len0 * a.x + len1 * b.x + len2 * c.x) / circum;
170   inCentreY = (len0 * a.y + len1 * b.y + len2 * c.y) / circum;
171 
172   return new jsts.geom.Coordinate(inCentreX, inCentreY);
173 };

176 /**
177  * Computes the centroid (centre of mass) of a triangle.
178  * This is also the point at which the triangle's three
179  * medians intersect (a triangle median is the segment from a vertex of the
180  * triangle to the midpoint of the opposite side).
181  * The centroid divides each median in a ratio of 2:1.
182  * The centroid always lies within the triangle.
183  *
184  *
185  * @param {jsts.geom.Coordinate}
186  *        a a vertx of the triangle.
187  * @param {jsts.geom.Coordinate}
188  *        b a vertx of the triangle.
189  * @param {jsts.geom.Coordinate}
190  *        c a vertx of the triangle.
191  * @return {jsts.geom.Coordinate}
192  *         the centroid of the triangle.
193  */
194 jsts.geom.Triangle.centroid = function(a, b, c) {
195   var x, y;
196 
197   x = (a.x + b.x + c.x) / 3;
198   y = (a.y + b.y + c.y) / 3;
199 
200   return new jsts.geom.Coordinate(x, y);
201 };

204 /**
205  * Computes the length of the longest side of a triangle
206  *
207  * @param {jsts.geom.Coordinate}
208  *        a a vertex of the triangle.
209  * @param {jsts.geom.Coordinate}
210  *        b a vertex of the triangle.
211  * @param {jsts.geom.Coordinate}
212  *        c a vertex of the triangle.
213  * @return {Number}
214  *         the length of the longest side of the triangle.
215  */
216 jsts.geom.Triangle.longestSideLength = function(a, b, c) {
217   var lenAB, lenBC, lenCA, maxLen;
218 
219   lenAB = a.distance(b);
220   lenBC = b.distance(c);
221   lenCA = c.distance(a);
222   maxLen = lenAB;
223 
224   if (lenBC > maxLen) {
225     maxLen = lenBC;
226   }
227   if (lenCA > maxLen) {
228     maxLen = lenCA;
229   }
230   return maxLen;
231 };

234 /**
235  * Computes the point at which the bisector of the angle ABC
236  * cuts the segment AC.
237  *
238  * @param {jsts.geom.Coordinate}
239  *        a a vertex of the triangle.
240  * @param {jsts.geom.Coordinate}
241  *        b a vertex of the triangle.
242  * @param {jsts.geom.Coordinate}
243  *        c a vertex of the triangle.
244  * @return {jsts.geom.Coordinate}
245  *         the angle bisector cut point.
246  */
247 jsts.geom.Triangle.angleBisector = function(a, b, c) {
248   /**
249    * Uses the fact that the lengths of the parts of the split segment
250    * are proportional to the lengths of the adjacent triangle sides
251    */
252   var len0, len2, frac, dx, dy, splitPt;
253 
254   len0 = b.distance(a);
255   len2 = b.distance(c);
256   frac = len0 / (len0 + len2);
257   dx = c.x - a.x;
258   dy = c.y - a.y;
259 
260   splitPt = new jsts.geom.Coordinate(a.x + frac * dx, a.y + frac * dy);
261   return splitPt;
262 };

265 /**
266  * Computes the 2D area of a triangle.
267  * The area value is always non-negative.
268  *
269  * @param {jsts.geom.Coordinate}
270  *        a a vertex of the triangle.
271  * @param {jsts.geom.Coordinate}
272  *        b a vertex of the triangle.
273  * @param {jsts.geom.Coordinate}
274  *        c a vertex of the triangle.
275  * @return {Number}
276  *         the area of the triangle.
277  */
278 jsts.geom.Triangle.area = function(a, b, c) {
279   return Math.abs(((c.x - a.x) * (b.y - a.y) -
280       (b.x - a.x) * (c.y - a.y)) / 2.0);
281 };

284 /**
285  * Computes the signed 2D area of a triangle.
286  * The area value is positive if the triangle is oriented CW,
287  * and negative if it is oriented CCW.
288  * <p>
289  * The signed area value can be used to determine point orientation, but
290  * the implementation in this method
291  * is susceptible to round-off errors.
292  * Use {@link CGAlgorithms#orientationIndex(Coordinate,
293  * Coordinate, Coordinate)} for robust orientation calculation.
294  *
295  * @param {jsts.geom.Coordinate}
296  *        a a vertex of the triangle.
297  * @param {jsts.geom.Coordinate}
298  *        b a vertex of the triangle.
299  * @param {jsts.geom.Coordinate}
300  *        c a vertex of the triangle.
301  * @return {Number}
302  *         the signed 2D area of the triangle.
303  *
304  */
305 jsts.geom.Triangle.signedArea = function(a, b, c) {
306   /**
307    * Uses the formula 1/2 * | u x v |
308    * where
309    *  u,v are the side vectors of the triangle
310    *  x is the vector cross-product
311    * For 2D vectors, this formula simplifies to the expression below
312    */
313   return ((c.x - a.x) * (b.y - a.y) - (b.x - a.x) * (c.y - a.y)) / 2.0;
314 };

338  * It is also the point at which the bisectors
