  7 /**
  8  * The purpose of this file is to facilitate OpenLayers compatibility by mixin
  9  * jsts.geom.* implementation into corresponding OpenLayers.Geometry.* then set
 10  * jsts.geom.* as an alias.
 11  *
 12  * Note that jsts.geom.Point is unified with jsts.geom.Coordinate.
 13  *
 14  * @requires jsts/geom/Coordinate.js
 15  * @requires jsts/geom/Point.js
 16  * @requires jsts/geom/GeometryCollection.js
 17  * @requires jsts/geom/MultiPoint.js
 18  * @requires jsts/geom/LineString.js
 19  * @requires jsts/geom/MultiLineString.js
 20  * @requires jsts/geom/LinearRing.js
 21  * @requires jsts/geom/Polygon.js
 22  * @requires jsts/geom/MultiPolygon.js
 23  */
 24 
 25 OpenLayers.Geometry.Point = OpenLayers.Class(OpenLayers.Geometry.Point,
 26     jsts.geom.Coordinate);
 27 jsts.geom.Coordinate = OpenLayers.Geometry.Point;
 28 jsts.geom.Point = OpenLayers.Class(jsts.geom.Point, jsts.geom.Coordinate);
 29 
 30 OpenLayers.Geometry.Collection = OpenLayers.Class(
 31     OpenLayers.Geometry.Collection, jsts.geom.GeometryCollection);
 32 jsts.geom.GeometryCollection = OpenLayers.Geometry.Collection;
 33 
 34 OpenLayers.Geometry.MultiPoint = OpenLayers.Class(
 35     OpenLayers.Geometry.MultiPoint, jsts.geom.MultiPoint);
 36 jsts.geom.MultiPoint = OpenLayers.Geometry.Point;
 37 
 38 OpenLayers.Geometry.LineString = OpenLayers.Class(
 39     OpenLayers.Geometry.LineString, jsts.geom.LineString);
 40 jsts.geom.LineString = OpenLayers.Geometry.LineString;
 41 
 42 OpenLayers.Geometry.MultiLineString = OpenLayers.Class(
 43     OpenLayers.Geometry.MultiLineString, jsts.geom.MultiLineString);
 44 jsts.geom.MultiLineString = OpenLayers.Geometry.MultiLineString;
 45 
 46 OpenLayers.Geometry.LinearRing = OpenLayers.Class(
 47     OpenLayers.Geometry.LinearRing, jsts.geom.LinearRing);
 48 jsts.geom.LinearRing = OpenLayers.Geometry.LinearRing;
 49 
 50 OpenLayers.Geometry.Polygon = OpenLayers.Class(OpenLayers.Geometry.Polygon,
 51     jsts.geom.Polygon);
 52 jsts.geom.Polygon = OpenLayers.Geometry.Polygon;
 53 
 54 OpenLayers.Geometry.MultiPolygon = OpenLayers.Class(
 55     OpenLayers.Geometry.MultiPolygon, jsts.geom.MultiPolygon);
 56 jsts.geom.MultiPolygon = OpenLayers.Geometry.MultiPolygon;
