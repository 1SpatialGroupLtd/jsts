  7 /**
  8  * The purpose of this file is to facilitate OpenLayers compatibility by mixin
  9  * jsts.geom.* implementation into corresponding OpenLayers.Geometry.* then set
 10  * jsts.geom.* as an alias.
 11  *
 12  * Note that jsts.geom.Point is unified with jsts.geom.Coordinate.
 13  *
 14  * @requires jsts/geom/Coordinate.js
 15  * @requires jsts/geom/Point.js
 16  * @requires jsts/geom/GeometryCollection.js
 17  * @requires jsts/geom/MultiPoint.js
 18  * @requires jsts/geom/LineString.js
 19  * @requires jsts/geom/MultiLineString.js
 20  * @requires jsts/geom/LinearRing.js
 21  * @requires jsts/geom/Polygon.js
 22  * @requires jsts/geom/MultiPolygon.js
 23  */
 24 
 25 
 26 /**
 27  * Extend OpenLayers.Geometry.Point with jsts.geom.Coordinate
 28  */
 29 OpenLayers.Geometry.Point = OpenLayers.Class(OpenLayers.Geometry.Point,
 30     jsts.geom.Coordinate);
 31 
 32 
 33 /**
 34  * Alias for extended OpenLayers.Geometry.Point
 35  */
 36 jsts.geom.Coordinate = OpenLayers.Geometry.Point;
 37 
 38 
 39 /**
 40  * Alias for jsts.geom.Coordinate
 41  */
