  9 /**
 10  * @param {Coordinate[]}
 11  *          pts
 12  * @param {Label}
 13  *          label
 14  * @augments jsts.geomgraph.GraphComponent
 15  * @constructor
 16  */
 17 jsts.geomgraph.Edge = function(pts, label) {
 18   this.pts = pts;
 19   this.label = label;
 20   this.eiList = new jsts.geomgraph.EdgeIntersectionList(this);
 21 };

 71 /**
 72  * Adds EdgeIntersections for one or both intersections found for a segment of
 73  * an edge to the edge intersection list.
 74  *
 75  * @param {LineIntersector}
 76  *          li
 77  * @param {int}
 78  *          segmentIndex
 79  * @param {int}
 80  *          geomIndex
 81  */
 82 jsts.geomgraph.Edge.prototype.addIntersections = function(li, segmentIndex,
 83     geomIndex) {
 84   for (var i = 0; i < li.getIntersectionNum(); i++) {
 85     this.addIntersection(li, segmentIndex, geomIndex, i);
 86   }
 87 };
 88 
 89 
 90 /**
 91  * Add an EdgeIntersection for intersection intIndex. An intersection that falls
 92  * exactly on a vertex of the edge is normalized to use the higher of the two
 93  * possible segmentIndexes
 94  *
 95  * @param {LineIntersector}
 96  *          li
 97  * @param {int}
 98  *          segmentIndex
 99  * @param {int}
100  *          geomIndex
101  * @param {int}
102  *          intIndex
103  */
104 jsts.geomgraph.Edge.prototype.addIntersection = function(li, segmentIndex,
105     geomIndex, intIndex) {
106   var intPt = new jsts.geom.Coordinate(li.getIntersection(intIndex));
107   var normalizedSegmentIndex = segmentIndex;
108   var dist = li.getEdgeDistance(geomIndex, intIndex);
109 
110   var nextSegIndex = normalizedSegmentIndex + 1;
111   if (nextSegIndex < this.pts.length) {
112     var nextPt = this.pts[nextSegIndex];
113 
114     // Normalize segment index if intPt falls on vertex
115     // The check for point equality is 2D only - Z values are ignored
116     if (intPt.equals2D(nextPt)) {
117       normalizedSegmentIndex = nextSegIndex;
118       dist = 0.0;
119     }
120   }
121   /**
122    * Add the intersection point to edge intersection list.
123    */
124   var ei = this.eiList.add(intPt, normalizedSegmentIndex, dist);
125 };
