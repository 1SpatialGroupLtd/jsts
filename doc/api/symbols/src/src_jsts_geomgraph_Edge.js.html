  9 /**
 10  * @param {Coordinate[]}

 12  * @param {Label}

 14  * @augments jsts.geomgraph.GraphComponent
 15  * @constructor
 16  */
 17 jsts.geomgraph.Edge = function(pts, label) {
 18   this.pts = pts;
 19   this.label = label;
 20   // this.mce = new MonotoneChainEdge();
 21   // this.depth = new Depth();
 22   this.eiList = new jsts.geomgraph.EdgeIntersectionList(this);
 23 };

 28 /**
 29  * Updates an IM from the label for an edge. Handles edges from both L and A
 30  * geometries.
 31  */
 32 jsts.geomgraph.Edge.updateIM = function(label, im) {
 33   im.setAtLeastIfValid(label.getLocation(0, jsts.geomgraph.Position.ON), label
 34       .getLocation(1, jsts.geomgraph.Position.ON), 1);
 35   if (label.isArea()) {
 36     im.setAtLeastIfValid(label.getLocation(0, jsts.geomgraph.Position.LEFT),
 37         label.getLocation(1, jsts.geomgraph.Position.LEFT), 2);
 38     im.setAtLeastIfValid(label.getLocation(0, jsts.geomgraph.Position.RIGHT),
 39         label.getLocation(1, jsts.geomgraph.Position.RIGHT), 2);
 40   }
 41 };

111 /**
112  * @param {int}
113  *          i
114  * @return {Coordinate}
115  */
116 jsts.geomgraph.Edge.prototype.getCoordinate = function(i) {
117   if (i === undefined) {
118     if (this.pts.length > 0) {
119       return this.pts[0];
120     } else {
121       return null;
122     }
123   }
124 
125   return this.pts[i];
126 };

145 /**
146  * Adds EdgeIntersections for one or both intersections found for a segment of
147  * an edge to the edge intersection list.
148  *
149  * @param {LineIntersector}
150  *          li
151  * @param {int}
152  *          segmentIndex
153  * @param {int}
154  *          geomIndex
155  */
156 jsts.geomgraph.Edge.prototype.addIntersections = function(li, segmentIndex,
157     geomIndex) {
158   for (var i = 0; i < li.getIntersectionNum(); i++) {
159     this.addIntersection(li, segmentIndex, geomIndex, i);
160   }
161 };
162 
163 
164 /**
165  * Add an EdgeIntersection for intersection intIndex. An intersection that falls
166  * exactly on a vertex of the edge is normalized to use the higher of the two
167  * possible segmentIndexes
168  *
169  * @param {LineIntersector}
170  *          li
171  * @param {int}
172  *          segmentIndex
173  * @param {int}
174  *          geomIndex
175  * @param {int}
176  *          intIndex
177  */
178 jsts.geomgraph.Edge.prototype.addIntersection = function(li, segmentIndex,
179     geomIndex, intIndex) {
180   var intPt = new jsts.geom.Coordinate(li.getIntersection(intIndex));
181   var normalizedSegmentIndex = segmentIndex;
182   var dist = li.getEdgeDistance(geomIndex, intIndex);
183   //normalize the intersection point location
184   var nextSegIndex = normalizedSegmentIndex + 1;
185   if (nextSegIndex < this.pts.length) {
186     var nextPt = this.pts[nextSegIndex];
187 
188     // Normalize segment index if intPt falls on vertex
189     // The check for point equality is 2D only - Z values are ignored
190     if (intPt.equals2D(nextPt)) {
191       normalizedSegmentIndex = nextSegIndex;
192       dist = 0.0;
193     }
194   }
195   /**
196    * Add the intersection point to edge intersection list.
197    */
198   var ei = this.eiList.add(intPt, normalizedSegmentIndex, dist);
199 };
