  9 /**
 10  * Models the end of an edge incident on a node.
 11  * EdgeEnds have a direction
 12  * determined by the direction of the ray from the initial
 13  * point to the next point.
 14  * EdgeEnds are comparable under the ordering
 15  * "a has a greater angle with the x-axis than b".
 16  * This ordering is used to sort EdgeEnds around a node.
 17  * @param {Edge} edge
 18  * @param {Coordinate} p0
 19  * @param {Coordinate} p1
 20  * @param {Label} label
 21  * @constructor
 22  */
 23 jsts.geomgraph.EdgeEnd = function(edge,  p0,  p1,  label) {
 24   this.edge = edge;
 25   if (p0 !== undefined) {
 26     if (p1 === undefined) {
 27       throw new jsts.error.IllegalArgumentError('expected two coords');
 28     }
 29     this.init(p0, p1);
 30     this.label = label;
 31   }
 32 };

 83 /**
 84    * @param {Coordinate} p0
 85    * @param {Coordinate} p1
 86    * @protected
 87    */
 88 jsts.geomgraph.EdgeEnd.prototype.init = function(p0,  p1)  {
 89   this.p0 = p0;
 90   this.p1 = p1;
 91   this.dx = p1.x - p0.x;
 92   this.dy = p1.y - p0.y;
 93   this.quadrant = jsts.geomgraph.Quadrant.quadrant(this.dx, this.dy);
 94   // TODO: Assert.isTrue(! (dx == 0 && dy == 0), 'EdgeEnd with identical endpoints found');
 95 };

121 /**
122    * Implements the total order relation:
123    * <p>
124    *    a has a greater angle with the positive x-axis than b
125    * <p>
126    * Using the obvious algorithm of simply computing the angle is not robust,
127    * since the angle calculation is obviously susceptible to roundoff.
128    * A robust algorithm is:
129    * - first compare the quadrant.  If the quadrants
130    * are different, it it trivial to determine which vector is "greater".
131    * - if the vectors lie in the same quadrant, the computeOrientation function
132    * can be used to decide the relative orientation of the vectors.
133    * @param {EdgeEnd} e
134    * @return {int}
135    */
136 jsts.geomgraph.EdgeEnd.prototype.compareDirection = function(e)  {
137   if (this.dx === e.dx && this.dy === e.dy)
138     return 0;
139   // if the rays are in different quadrants, determining the ordering is trivial
140   if (this.quadrant > e.quadrant) return 1;
141   if (this.quadrant < e.quadrant) return -1;
142   // vectors are in the same quadrant - check relative orientation of direction vectors
143   // this is > e if it is CCW of e
144   return jsts.algorithm.CGAlgorithms.computeOrientation(e.p0, e.p1, this.p1);
145 };
