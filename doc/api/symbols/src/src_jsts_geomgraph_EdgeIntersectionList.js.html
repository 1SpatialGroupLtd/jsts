 12 jsts.geomgraph.EdgeIntersectionList = function(edge) {

 27 /**
 28  * Adds an intersection into the list, if it isn't already there.
 29  * The input segmentIndex and dist are expected to be normalized.
 30  * @param {Coordinate} intPt
 31  * @param {int} segmentIndex
 32  * @param {double} dist
 33  * @return {EdgeIntersection} the EdgeIntersection found or added.
 34  */
 35 jsts.geomgraph.EdgeIntersectionList.prototype.add = function(intPt, segmentIndex, dist)
 36     {
 37   var eiNew = new jsts.geomgraph.EdgeIntersection(intPt, segmentIndex, dist);
 38   var ei = this.nodeMap.get(eiNew);
 39   if (ei != null) {
 40     return ei;
 41   }
 42   this.nodeMap.put(eiNew, eiNew);
 43   return eiNew;
 44 };
