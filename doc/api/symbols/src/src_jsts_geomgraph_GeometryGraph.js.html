 12 /**
 13  * A GeometryGraph is a graph that models a given Geometry
 14  *
 15  * @param {int}
 16  *          argIndex
 17  * @param {Geometry}
 18  *          parentGeom
 19  * @param {BoundaryNodeRule}
 20  *          boundaryNodeRule
 21  * @augments jsts.planargraph.PlanarGraph
 22  */
 23 jsts.geomgraph.GeometryGraph = function(argIndex, parentGeom, boundaryNodeRule) {
 24   jsts.geomgraph.GeometryGraph.prototype.constructor.call(this);
 25 
 26   this.lineEdgeMap = new jsts.Hashtable();
 27   this.ptLocator = new jsts.algorithm.PointLocator();
 28 
 29   this.argIndex = argIndex;
 30   this.parentGeom = parentGeom;
 31   this.boundaryNodeRule = boundaryNodeRule ? boundaryNodeRule
 32       : jsts.algorithm.BoundaryNodeRule.OGC_SFS_BOUNDARY_RULE;
 33   if (parentGeom != null) {
 34     this.add(parentGeom);
 35   }
 36 };
 37 
 38 jsts.geomgraph.GeometryGraph.prototype = new jsts.geomgraph.PlanarGraph();

 41 /**
 42  * @param {BoundaryNodeRule}
 43  *          boundaryNodeRule
 44  * @param {int}
 45  *          boundaryCount
 46  * @return {int}
 47  */
 48 jsts.geomgraph.GeometryGraph.determineBoundary = function(boundaryNodeRule,
 49     boundaryCount) {
 50   return boundaryNodeRule.isInBoundary(boundaryCount) ? Location.BOUNDARY
 51       : Location.INTERIOR;
 52 };

147 /**
148  * @param {Geometry}
149  *          g
150  */
151 jsts.geomgraph.GeometryGraph.prototype.add = function(g) {
152   if (g.isEmpty()) {
153     return;
154   }
155 
156   // check if this Geometry should obey the Boundary Determination Rule
157   // all collections except MultiPolygons obey the rule
158   if (g instanceof jsts.geom.MultiPolygon)
159     useBoundaryDeterminationRule = false;
160 
161   if (g instanceof jsts.geom.Polygon)
162     this.addPolygon(g);
163   // LineString also handles LinearRings
164   else if (g instanceof jsts.geom.LineString)
165     this.addLineString(g);
166   else if (g instanceof jsts.geom.Point)
167     this.addPoint(g);
168   else if (g instanceof jsts.geom.MultiPoint)
169     this.addCollection(g);
170   else if (g instanceof jsts.geom.MultiLineString)
171     this.addCollection(g);
172   else if (g instanceof jsts.geom.MultiPolygon)
173     this.addCollection(g);
174   else if (g instanceof jsts.geom.GeometryCollection)
175     this.addCollection(g);
176   else
177     throw new jsts.error.IllegalArgumentError('Geometry type not supported.');
178 };

181 /**
182  * @param {LineString}
183  *          line
184  * @private
185  */
186 jsts.geomgraph.GeometryGraph.prototype.addLineString = function(line) {
187   var coords = jsts.geom.CoordinateArrays.removeRepeatedPoints(line
188       .getCoordinates());
189 
190   if (coords.length < 2) {
191     hasTooFewPoints = true;
192     invalidPoint = coord[0];
193     return;
194   }
195 
196   // add the edge for the LineString
197   // line edges do not have locations for their left and right sides
198   var e = new jsts.geomgraph.Edge(coords, new jsts.geomgraph.Label(
199       this.argIndex, jsts.geom.Location.INTERIOR));
200   this.lineEdgeMap.put(line, e);
201   this.insertEdge(e);
202   /**
203    * Add the boundary points of the LineString, if any. Even if the LineString
204    * is closed, add both points as if they were endpoints. This allows for the
205    * case that the node already exists and is a boundary point.
206    */
207   if (coords.length >= 2 === false) {
208     throw new jsts.error.IllegalArgumentError(
209         'found LineString with single point');
210   }
211 
212   this.insertBoundaryPoint(this.argIndex, coords[0]);
213   this.insertBoundaryPoint(this.argIndex, coords[coords.length - 1]);
214 };

217 /**
218  * Compute self-nodes, taking advantage of the Geometry type to minimize the
219  * number of intersection tests. (E.g. rings are not tested for
220  * self-intersection, since they are assumed to be valid).
221  *
222  * @param {LineIntersector}
223  *          li the LineIntersector to use.
224  * @param {boolean}
225  *          computeRingSelfNodes if <false>, intersection checks are optimized
226  *          to not test rings for self-intersection.
227  * @return {SegmentIntersector} the SegmentIntersector used, containing
228  *         information about the intersections found.
229  */
230 jsts.geomgraph.GeometryGraph.prototype.computeSelfNodes = function(li,
231     computeRingSelfNodes) {
232   var si = new jsts.geomgraph.index.SegmentIntersector(li, true, false);
233   var esi = this.createEdgeSetIntersector();
234   // optimized test for Polygons and Rings
235   if (!computeRingSelfNodes &&
236       (parentGeom instanceof LinearRing || parentGeom instanceof Polygon || parentGeom instanceof MultiPolygon)) {
237     esi.computeIntersections(this.edges, si, false);
238   } else {
239     esi.computeIntersections(this.edges, si, true);
240   }
241   this.addSelfIntersectionNodes(this.argIndex);
242   return si;
243 };
