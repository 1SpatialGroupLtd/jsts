  8 /**
  9  * A GeometryGraph is a graph that models a given Geometry
 10  *
 11  * @param {int}
 12  *          argIndex
 13  * @param {Geometry}
 14  *          parentGeom
 15  * @param {BoundaryNodeRule}
 16  *          boundaryNodeRule
 17  * @augments jsts.planargraph.PlanarGraph
 18  */
 19 jsts.geomgraph.GeometryGraph = function(argIndex, parentGeom, boundaryNodeRule) {
 20   this.lineEdgeMap = {};
 21   this.ptLocator = new jsts.algorithm.PointLocator();
 22 
 23   this.argIndex = argIndex;
 24   this.parentGeom = parentGeom;
 25   this.boundaryNodeRule = boundaryNodeRule ? boundaryNodeRule
 26       : jsts.algorithm.BoundaryNodeRule.OGC_SFS_BOUNDARY_RULE;
 27   if (parentGeom != null) {
 28     this.add(parentGeom);
 29   }
 30 };
 31 
 32 jsts.geomgraph.GeometryGraph.prototype = new jsts.geomgraph.PlanarGraph();
 33

 35 /**
 36  * @param {BoundaryNodeRule}
 37  *          boundaryNodeRule
 38  * @param {int}
 39  *          boundaryCount
 40  * @return {int}
 41  */
 42 jsts.geomgraph.GeometryGraph.determineBoundary = function(boundaryNodeRule,
 43     boundaryCount) {
 44   return boundaryNodeRule.isInBoundary(boundaryCount) ? Location.BOUNDARY
 45       : Location.INTERIOR;
 46 };

151 /**
152  * Compute self-nodes, taking advantage of the Geometry type to minimize the
153  * number of intersection tests. (E.g. rings are not tested for
154  * self-intersection, since they are assumed to be valid).
155  *
156  * @param {LineIntersector}
157  *          li the LineIntersector to use.
158  * @param {boolean}
159  *          computeRingSelfNodes if <false>, intersection checks are optimized
160  *          to not test rings for self-intersection.
161  * @return {SegmentIntersector} the SegmentIntersector used, containing
162  *         information about the intersections found.
163  */
164 jsts.geomgraph.GeometryGraph.prototype.computeSelfNodes = function(li,
165     computeRingSelfNodes) {
166   var si = new jsts.geomgraph.index.SegmentIntersector(li, true, false);
167   var esi = this.createEdgeSetIntersector();
168   // optimized test for Polygons and Rings
169   if (!computeRingSelfNodes &&
170       (parentGeom instanceof LinearRing || parentGeom instanceof Polygon || parentGeom instanceof MultiPolygon)) {
171     esi.computeIntersections(edges, si, false);
172   } else {
173     esi.computeIntersections(edges, si, true);
174   }
175   this.addSelfIntersectionNodes(argIndex);
176   return si;
177 };
