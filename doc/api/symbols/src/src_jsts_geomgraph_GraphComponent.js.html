158 /**
159  * Update the IM with the contribution for this component. A component only
160  * contributes if it has a labelling for both parent geometries
161  *
162  * @param {IntersectionMatrix}
163  *          im
164  */
165 jsts.geomgraph.GraphComponent.prototype.updateIM = function(im) {
166   if (this.label.getGeometryCount() >= 2) {
167     throw new jsts.error.NotRepresentableException('found partial label');
168   }
169   this.computeIM(im);
170 };
