  9 /**
 10  * @constructor
 11  * @augments jsts.geomgraph.GraphComponent
 12  */
 13 jsts.geomgraph.Node = function(coord, edges) {
 14   this.coord = coord;
 15   this.edges = edges;
 16   this.label = new jsts.geomgraph.Label(0, jsts.geom.Location.NONE);
 17 };
 18 
 19 jsts.geomgraph.Node.prototype = new jsts.geomgraph.GraphComponent();
 20 
 21 
 22 /**
 23  * only non-null if this node is precise
 24  */
 25 jsts.geomgraph.Node.prototype.coord = null;
 26 jsts.geomgraph.Node.prototype.edges = null;
 27 
 28 jsts.geomgraph.Node.prototype.isIsolated = function() {
 29   return (this.label.getGeometryCount() == 1);
 30 };
 31 
 32 jsts.geomgraph.Node.prototype.setLabel2 = function(argIndex,  onLocation) {
 33   if (this.label === null) {
 34     this.label = new jsts.geomgraph.Label(argIndex, onLocation);
 35   }
 36   else
 37     this.label.setLocation(argIndex, onLocation);
 38 };
 39 
 40 
 41 /**
 42  * Updates the label of a node to BOUNDARY,
 43  * obeying the mod-2 boundaryDetermination rule.
 44  */
 45 jsts.geomgraph.Node.prototype.setLabelBoundary = function(argIndex) {
 46   // determine the current location for the point (if any)
 47   var loc = jsts.geom.Location.NONE;
 48   if (this.label !== null)
 49     loc = this.label.getLocation(argIndex);
 50   // flip the loc
 51   var newLoc;
 52   switch (loc) {
 53     case jsts.geom.Location.BOUNDARY: newLoc = jsts.geom.Location.INTERIOR; break;
 54     case jsts.geom.Location.INTERIOR: newLoc = jsts.geom.Location.BOUNDARY; break;
 55     default: newLoc = jsts.geom.Location.BOUNDARY; break;
 56   }
 57   this.label.setLocation(argIndex, newLoc);
 58 };

 61 /**
 62  * Add the edge to the list of edges at this node
 63  */
 64 jsts.geomgraph.Node.prototype.add = function(e)
 65     {
 66   // Assert: start pt of e is equal to node point
 67   this.edges.insert(e);
 68   e.setNode(this);
 69 };
 70 
 71 jsts.geomgraph.Node.prototype.getCoordinate = function() {
 72   return this.coord;
 73 };
 74 jsts.geomgraph.Node.prototype.getEdges = function() {
 75   return this.edges;
 76 };
