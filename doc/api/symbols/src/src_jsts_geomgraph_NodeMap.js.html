  9 /**
 10  * A map of nodes, indexed by the coordinate of the node
 11  *
 12  * @constructor
 13  */
 14 jsts.geomgraph.NodeMap = function() {
 15   this.nodeMap = new jsts.Hashtable();
 16 };
 17 
 18 jsts.geomgraph.NodeMap.prototype.nodeMap = null;
 19 
 20 
 21 /**
 22  * This method expects that a node has a coordinate value.
 23  *
 24  * @param {Coordinate/Node}
 25  *          arg
 26  * @return {Node}
 27  */
 28 jsts.geomgraph.NodeMap.prototype.addNode = function(arg) {
 29   var node;
 30 
 31   if (arg instanceof jsts.geom.Coordinate) {
 32     var coord = arg;
 33     node = this.nodeMap.get(coord);
 34     if (node == null) {
 35       node = new jsts.geomgraph.Node(coord, null);
 36       this.nodeMap.put(coord, node);
 37     }
 38     return node;
 39   } else if (arg instanceof jsts.geomgraph.Node) {
 40     var node = arg;
 41     node = nodeMap.get(n.getCoordinate());
 42     if (node === null) {
 43       this.nodeMap.put(n.getCoordinate(), n);
 44       return n;
 45     }
 46     node.mergeLabel(n);
 47     return node;
 48   }
 49 };
 50 
 51 
 52 /**
 53  * Adds a node for the start point of this EdgeEnd (if one does not already
 54  * exist in this map). Adds the EdgeEnd to the (possibly new) node.
 55  *
 56  * @param {EdgeEnd}
 57  *          e
 58  */
 59 jsts.geomgraph.NodeMap.prototype.add = function(e) {
 60   var p = e.getCoordinate();
 61   var n = addNode(p);
 62   n.add(e);
 63 };
 64 
 65 
 66 /**
 67  * @param {Coordinate}
 68  *          coord
 69  * @return {Node} the node if found; null otherwise.
 70  */
 71 jsts.geomgraph.NodeMap.prototype.find = function(coord) {
 72   return nodeMap.get(coord);
 73 };
 74 
 75 
 76 /**
 77  * @return {Node[]}
 78  */
 79 jsts.geomgraph.NodeMap.prototype.values = function() {
 80   // TODO: sort by key?
 81   return this.nodeMap.values();
 82 };
 83 
 84 
 85 /**
 86  * @param {int}
 87  *          geomIndex
 88  * @return {Node[]}
 89  */
 90 jsts.geomgraph.NodeMap.prototype.getBoundaryNodes = function(geomIndex) {
 91   var bdyNodes = [];
 92   var values = this.values();
 93   for (var i = 0; i < values.length; i++) {
 94     var node = values[i];
 95     if (node.getLabel().getLocation(geomIndex) === jsts.geom.Location.BOUNDARY)
 96       bdyNodes.add(node);
 97   }
 98   return bdyNodes;
 99 };
