 14 /**
 15  * The computation of the <code>IntersectionMatrix</code> relies on the use of
 16  * a structure called a "topology graph". The topology graph contains nodes and
 17  * edges corresponding to the nodes and line segments of a <code>Geometry</code>.
 18  * Each node and edge in the graph is labeled with its topological location
 19  * relative to the source geometry.
 20  * <P>
 21  * Note that there is no requirement that points of self-intersection be a
 22  * vertex. Thus to obtain a correct topology graph, <code>Geometry</code>s
 23  * must be self-noded before constructing their graphs.
 24  * <P>
 25  * Two fundamental operations are supported by topology graphs:
 26  * <UL>
 27  * <LI>Computing the intersections between all the edges and nodes of a single
 28  * graph
 29  * <LI>Computing the intersections between the edges and nodes of two different
 30  * graphs
 31  * </UL>
 32  *
 33  * @constructor
 34  */
 35 jsts.geomgraph.PlanarGraph = function(nodeFactory) {
 36   this.edges = [];
 37   this.nodes = new jsts.geomgraph.NodeMap(nodeFactory || new jsts.geomgraph.NodeFactory());
 38 };
 39 
 40 
 41 /**
 42  * @protected
 43  */
 44 jsts.geomgraph.PlanarGraph.prototype.edges = null;
 45 
 46 
 47 /**
 48  * @type {jsts.geomgraph.NodeMap}
 49  */
 50 jsts.geomgraph.PlanarGraph.prototype.nodes = null;
 51 
 52 jsts.geomgraph.PlanarGraph.prototype.isBoundaryNode = function(geomIndex, coord) {
 53   var node = this.nodes.find(coord);
 54   if (node == null) return false;
 55   var label = node.getLabel();
 56   if (label !== null && label.getLocation(geomIndex) === jsts.geom.Location.BOUNDARY) return true;
 57   return false;
 58 };
 59 
 60 jsts.geomgraph.PlanarGraph.prototype.insertEdge = function(e) {
 61   this.edges.push(e);
 62 };
 63 
 64 jsts.geomgraph.PlanarGraph.prototype.getEdges = function() {
 65   return this.edges;
 66 };
 67 
 68 jsts.geomgraph.PlanarGraph.prototype.getNodes = function() {
 69   return this.nodes.values();
 70 };
