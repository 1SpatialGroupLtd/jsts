  9 /**
 10  * Computes the intersection of line segments, and adds the intersection to the
 11  * edges containing the segments.
 12  *
 13  * @param {LineIntersector}
 14  *          li
 15  * @param {boolean}
 16  *          includeProper
 17  * @param {boolean}
 18  *          recordIsolated
 19  * @constructor
 20  */
 21 jsts.geomgraph.index.SegmentIntersector = function(li, includeProper,
 22     recordIsolated) {
 23   this.li = li;
 24   this.includeProper = includeProper;
 25   this.recordIsolated = recordIsolated;
 26 };
 27 
 28 
 29 /**
 30  * @param {int}
 31  *          i1
 32  * @param {int}
 33  *          i2
 34  * @return {boolean}
 35  */
 36 jsts.geomgraph.index.SegmentIntersector.isAdjacentSegments = function(i1, i2) {
 37   return Math.abs(i1 - i2) === 1;
 38 };
