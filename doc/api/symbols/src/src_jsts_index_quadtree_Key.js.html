  9 /**
 10  * A Key is a unique identifier for a node in a quadtree. It contains a
 11  * lower-left point and a level number. The level number is the power of two for
 12  * the size of the node envelope.
 13  *
 14  * @constructor
 15  */
 16 jsts.index.quadtree.Key = OpenLayers.Class();
 17 
 18 
 19 /**
 20  * Computes the quad-level for specified envelope
 21  *
 22  * @param {jsts.geom.Envelope}
 23  *          env the envelope to calculate level for.
 24  * @return {Number} The calculated level.
 25  */
 26 jsts.index.quadtree.Key.computeQuadLevel = function(env) {
 27   var dx, dy, dMax, level;
 28 
 29   dx = env.getWidth();
 30   dy = env.getHeight();
 31   dMax = dx > dy ? dx : dy;
 32   level = jsts.index.DoubleBits.exponent(dMax) + 1;
 33   return level;
 34 };

 37 /**
 38  * Initializes a new key
 39  *
 40  * @param {jsts.geom.Envelope}
 41  *          itemEnv the envelope of the key.
 42  */
 43 jsts.index.quadtree.Key.prototype.initialize = function(itemEnv) {
 44   // the fields which make up the key
 45   this.pt = new jsts.geom.Coordinate();
 46   this.level = 0;
 47   // auxiliary data which is derived from the key for use in computation
 48   this.env = null;
 49 
 50   this.computeKey(itemEnv);
 51 };

 84 /**
 85  * Gets the centre of this key
 86  *
 87  * @return {jsts.geom.Coordinate} the center-point.
 88  */
 89 jsts.index.quadtree.Key.prototype.getCentre = function() {
 90   var x, y;
 91   x = (this.env.getMinX() + this.env.getMaxX()) / 2;
 92   y = (this.env.getMinY() + this.env.getMaxY()) / 2;
 93   return new jsts.geom.Coordinate(x, y);
 94 };

 97 /**
 98  * Will call appropriate computeKey* method depending on arguments.
 99  */
100 jsts.index.quadtree.Key.prototype.computeKey = function() {
101   if (arguments[0] instanceof jsts.geom.Envelope) {
102     this.computeKeyFromEnvelope(arguments[0]);
103   } else {
104     this.computeKeyFromLevel(arguments[0], arguments[1]);
105   }
106 };
107 
108 
109 /**
110  * Computes the key from specified envlope.
111  *
112  * @param {jsts.geom.Envelope}
113  *          env the envelope.
114  */
115 jsts.index.quadtree.Key.prototype.computeKeyFromEnvelope = function(env) {
116   this.level = jsts.index.quadtree.Key.computeQuadLevel(env);
117   this.env = new jsts.geom.Envelope();
118   this.computeKey(this.level, env);
119   while (!this.env.contains(env)) {
120     this.level += 1;
121     this.computeKey(this.level, env);
122   }
123 };
124 
125 
126 /**
127  * Computes a key from a level and an envelope
128  *
129  * @param {Number}
130  *          level the level.
131  * @param {jsts.geom.Envelope}
132  *          env the envelope.
133  */
134 jsts.index.quadtree.Key.prototype.computeKeyFromLevel = function(level, env) {
135   var quadSize = jsts.index.DoubleBits.powerOf2(level);
136   this.pt.x = Math.floor(env.getMinX() / quadSize) * quadSize;
137   this.pt.y = Math.floor(env.getMinY() / quadSize) * quadSize;
138   this.env.init(this.pt.x, this.pt.x + quadSize, this.pt.y, this.pt.y +
139       quadSize);
140 };
