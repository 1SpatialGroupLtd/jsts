  8 /**
  9  * Represents a node of a {@link Quadtree}. Nodes contain items which have a
 10  * spatial extent corresponding to the node's position in the quadtree.
 11  *
 12  */
 13 jsts.index.quadtree.Node = OpenLayers.Class(jsts.index.quadtree.NodeBase);
 14 
 15 
 16 /**
 17  * Creates a node from specified envelope
 18  *
 19  * @param {jsts.geom.Envelope}
 20  *          env the envelope.
 21  * @return {jsts.index.quadtree.Node} the created node.
 22  */
 23 jsts.index.quadtree.Node.prototype.createNode = function(env) {
 24   var key, node;
 25   key = new jsts.index.quadtree.Key(env);
 26   node = new jsts.index.quadtree.Node(key.getEnvelope(), key.getLevel());
 27 
 28   return node;
 29 };

 32 /**
 33  * Creates an expanded node.
 34  *
 35  * @param {jsts.index.quadtree.Node}
 36  *          node the node to create a new node from.
 37  * @param {jsts.geom.Envelope}
 38  *          addEnv the envelope.
 39  * @return {jsts.index.quadtree.Node} the created node.
 40  */
 41 jsts.index.quadtree.Node.prototype.createExpanded = function(node, addEnv) {
 42   var expandEnv = new jsts.geom.Envelope(addEnv), largerNode;
 43 
 44   if (node !== null) {
 45     expandEnv.expandToInclude(node.env);
 46   }
 47 
 48   largerNode = jsts.index.quadtree.Node.prototype.createNode(expandEnv);
 49   if (node !== null) {
 50     largerNode.insertNode(node);
 51   }
 52 
 53   return largerNode;
 54 };

 57 /**
 58  * Initilalizes a node using the inheritance mechanism provided by OpenLayers.
 59  *
 60  * @param {jsts.geom.Envelope}
 61  *          env Envelope to initialize from.
 62  * @param {Number}
 63  *          level The level.
 64  */
 65 jsts.index.quadtree.Node.prototype.initialize = function(env, level) {
 66   jsts.index.quadtree.NodeBase.prototype.initialize.apply(this, arguments);
 67 
 68   this.env = env;
 69   this.level = level;
 70   this.centre = new jsts.geom.Coordinate();
 71   this.centre.x = (env.getMinX() + env.getMaxX()) / 2;
 72   this.centre.y = (env.getMinY() + env.getMaxY()) / 2;
 73 };

 77  * Gets the envelope for this node

 79  * @return {jsts.geom.Envelope} the envelope.

 81 jsts.index.quadtree.Node.prototype.getEnvelope = function() {

 86 /**
 87  * Checks wheter the provided envelope intersects this nodes envelope.
 88  *
 89  * @param {jsts.geom.Envelope}
 90  *          searchEnv the envelope to search.
 91  * @return {Boolean} True if searchEnv intersects this nodes envelope.
 92  */
 93 jsts.index.quadtree.Node.prototype.isSearchMatch = function(searchEnv) {
 94   return this.env.intersects(searchEnv);
 95 };

 98 /**
 99  * Returns the subquad containing the envelope. Creates the subquad if it does
100  * not already exist.
101  *
102  * @param {jsts.geom.Envelope}
103  *          searchEnv the input envelope.
104  * @return {jsts.index.quadtree.Node} the node containing the searchEnv.
105  */
106 jsts.index.quadtree.Node.prototype.getNode = function(searchEnv) {
107   var subnodeIndex = this.getSubnodeIndex(searchEnv, this.centre), node;
108 
109   // if subquadIndex is -1 searchEnv is not contained in a subquad
110   if (subnodeIndex !== -1) {
111     // create the quad if it does not exist
112     node = this.getSubnode(subnodeIndex);
113     // recursively search the found/created quad
114     return node.getNode(searchEnv);
115   } else {
116     return this;
117   }
118 };
119 
120 
121 /**
122  * Returns the smallest <i>existing</i> node containing the envelope.
123  *
124  * @param {jsts.geom.Envelope}
125  *          searchEnv input Envelope.
126  * @return {jsts.index.quadtree.Node} the smallest node containing searchEnv.
127  */
128 jsts.index.quadtree.Node.prototype.find = function(searchEnv) {
129   var subnodeIndex = this.getSubnodeIndex(searchEnv, this.centre), node;
130   if (subnodeIndex === -1) {
131     return this;
132   }
133 
134   if (this.subnode[subnodeIndex] !== null) {
135     // query lies in subquad, so search it
136     node = this.subnode[subnodeIndex];
137     return node.find(searchEnv);
138   }
139 
140   // no existing subquad, so return this one anyway
141   return this;
142 };

145 /**
146  * Inserts a child-node
147  *
148  * @param {jsts.index.quadtree.Node}
149  *          node to insert.
150  */
151 jsts.index.quadtree.Node.prototype.insertNode = function(node) {
152   var index = this.getSubnodeIndex(node.env, this.centre), childNode;
153   if (node.level === this.level - 1) {
154     this.subnode[index] = node;
155   } else {
156     // the quad is not a direct child, so make a new child quad to contain it
157     // and recursively insert the quad
158     childNode = this.createSubnode(index);
159     childNode.insertNode(node);
160     this.subnode[index] = childNode;
161   }
162 };

165 /**
166  * get the subquad for the index. If it doesn't exist, create it
167  *
168  * @param {Number}
169  *          index the index of the subnode to get.
170  * @return {jsts.index.quadtree.Node} the specified subnode.
171  */
172 jsts.index.quadtree.Node.prototype.getSubnode = function(index) {
173   if (this.subnode[index] === null) {
174     this.subnode[index].createSubnode(index);
175   }
176   return this.subnode[index];
177 };

180 /**
181  * Creates a subnode
182  *
183  * @param {Number}
184  *          index The index (0-4) on where to create a subnode.
185  * @return {jsts.index.quadtree.Node} the created node.
186  */
187 jsts.index.quadtree.Node.prototype.createSubnode = function(index) {
188   var minx = 0.0, maxx = 0.0, miny = 0.0, maxy = 0.0, sqEnv, node;
189   // create a new subquad in the appropriate quadrant
190   switch (index) {
191     case 0:
192       minx = this.env.getMinX();
193       maxx = this.centre.x;
194       miny = this.env.getMinY();
195       maxy = this.centre.y;
196       break;
197     case 1:
198       minx = this.centre.x;
199       maxx = this.env.getMaxX();
200       miny = this.env.getMinY();
201       maxy = this.centre.y;
202       break;
203     case 2:
204       minx = this.env.getMinX();
205       maxx = this.centre.x;
206       miny = this.centre.y;
207       maxy = this.env.getMaxY();
208       break;
209     case 3:
210       minx = this.centre.x;
211       maxx = this.env.getMaxX();
212       miny = this.centre.y;
213       maxy = this.env.getMaxY();
214       break;
215   }
216 
217   sqEnv = new jsts.geom.Envelope(minx, maxx, miny, maxy);
218   node = new jsts.index.quadtree.Node(sqEnv, this.level - 1);
219 
220   return node;
221 };
