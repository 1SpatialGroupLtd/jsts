  3 /**
  4  * QuadRoot is the root of a single Quadtree. It is centred at the origin, and
  5  * does not have a defined extent.
  6  *
  7  */
  8 jsts.index.quadtree.Root = OpenLayers.Class(jsts.index.quadtree.NodeBase);
  9 
 10 
 11 /**
 12  * Initilalizes a root using the inheritance mechanism provided by OpenLayers.
 13  */
 14 jsts.index.quadtree.Root.prototype.initialize = function() {
 15   jsts.index.quadtree.NodeBase.prototype.initialize.apply(this, arguments);
 16 
 17   // the root quad is centred at the origin.
 18   this.origin = new jsts.geom.Coordinate(0.0, 0.0);
 19 };
 20 
 21 
 22 /**
 23  * Insert an item into the quadtree this is the root of.
 24  *
 25  * @param {jsts.geom.Envelope}
 26  *          itemEnv the item envelope.
 27  * @param {Object}
 28  *          item the item to insert.
 29  */
 30 jsts.index.quadtree.Root.prototype.insert = function(itemEnv, item) {
 31   var index = this.getSubnodeIndex(itemEnv, this.origin);
 32 
 33   // if index is -1, itemEnv must cross the X or Y axis.
 34   if (index === -1) {
 35     this.add(item);
 36     return;
 37   }
 38   /**
 39    * the item must be contained in one quadrant, so insert it into the tree for
 40    * that quadrant (which may not yet exist)
 41    */
 42   var node = this.subnode[index];
 43   /**
 44    * If the subquad doesn't exist or this item is not contained in it, have to
 45    * expand the tree upward to contain the item.
 46    */
 47 
 48   if (node === null || !node.getEnvelope().contains(itemEnv)) {
 49     var largerNode = node.createExpanded(node, itemEnv);
 50     this.subnode[index] = largerNode;
 51   }
 52   /**
 53    * At this point we have a subquad which exists and must contain contains the
 54    * env for the item. Insert the item into the tree.
 55    */
 56   this.insertContained(this.subnode[index], itemEnv, item);
 57 };

 60 /**
 61  * insert an item which is known to be contained in the tree rooted at the given
 62  * QuadNode root. Lower levels of the tree will be created if necessary to hold
 63  * the item.
 64  *
 65  * @param {jsts.index.quadtree.Node}
 66  *          tree the root-node of the tree.
 67  * @param {jsts.geom.Envelope}
 68  *          itemEnv the envelope.
 69  * @param {Object}
 70  *          item the item to insert.
 71  */
 72 jsts.index.quadtree.Root.prototype.insertContained = function(tree, itemEnv,
 73     item) {
 74   /**
 75    * Do NOT create a new quad for zero-area envelopes - this would lead to
 76    * infinite recursion. Instead, use a heuristic of simply returning the
 77    * smallest existing quad containing the query
 78    */
 79   var isZeroX, isZeroY, node;
 80   isZeroX = jsts.index.IntervalSize.prototype.isZeroWidth(itemEnv.getMinX(),
 81       itemEnv.getMaxX());
 82   isZeroY = jsts.index.IntervalSize.prototype.isZeroWidth(itemEnv.getMinY(),
 83       itemEnv.getMaxY());
 84 
 85   if (isZeroX || isZeroY) {
 86     node = tree.find(itemEnv);
 87   } else {
 88     node = tree.getNode(itemEnv);
 89   }
 90   node.add(item);
 91 };
