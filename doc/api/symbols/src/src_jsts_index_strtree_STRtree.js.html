  7 /**
  8  *  A query-only R-tree created using the Sort-Tile-Recursive (STR) algorithm.
  9  *  For two-dimensional spatial data.
 10  * <P>
 11  *  The STR packed R-tree is simple to implement and maximizes space
 12  *  utilization; that is, as many leaves as possible are filled to capacity.
 13  *  Overlap between nodes is far less than in a basic R-tree. However, once the
 14  *  tree has been built (explicitly or on the first call to #query), items may
 15  *  not be added or removed.
 16  * <P>
 17  * Described in: P. Rigaux, Michel Scholl and Agnes Voisard.
 18  * <i>Spatial Databases With Application To GIS</i>.
 19  * Morgan Kaufmann, San Francisco, 2002.
 20  *
 21  * @version 1.7
 22  */
 23

 26 /**
 27  * Constructs an STRtree with the default node capacity or with the
 28  * given maximum number of child nodes that a node may have.
 29  * <p>
 30  * The minimum recommended capacity setting is 4.
 31  *
 32  *
 33  * @param {number} [nodeCapacity].
 34  * @extends {jsts.index.strtree.AbstractSTRtree}
 35  * @extends {jsts.index.SpatialIndex}
 36  * @constructor
 37  */
 38 jsts.index.strtree.STRtree = function(nodeCapacity) {
 39   nodeCapacity = nodeCapacity || jsts.index.strtree.STRtree.DEFAULT_NODE_CAPACITY;
 40   jsts.index.strtree.AbstractSTRtree.call(this, nodeCapacity);
 41 };
 42 
 43 OpenLayers.inherit(jsts.index.strtree.STRtree,
 44                    jsts.index.strtree.AbstractSTRtree,
 45                    jsts.index.SpatialIndex);

 48 /**
 49  * @type {Object} implements function for comparison
 50  * @private
 51  */
 52 jsts.index.strtree.STRtree.prototype.xComparator = {
 53   compare: function(o1, o2) {
 54     return jsts.index.strtree.AbstractSTRtree.prototype.compareDoubles(
 55         jsts.index.strtree.STRtree.prototype.centreX(o1.getBounds()),
 56         jsts.index.strtree.STRtree.prototype.centreX(o2.getBounds())
 57     );
 58   }
 59 };

 76 /**
 77  * @param {jsts.geom.Envelope} e
 78  * @return {number}
 79  */
 80 jsts.index.strtree.STRtree.prototype.centreX = function(e) {
 81   return jsts.index.strtree.STRtree.prototype.avg(e.getMinX(), e.getMaxX());
 82 };
 83 
 84 
 85 /**
 86  * @param {jsts.geom.Envelope} e
 87  * @return {number}
 88  */
 89 jsts.index.strtree.STRtree.prototype.centreY = function(e) {
 90   return jsts.index.strtree.STRtree.prototype.avg(e.getMinY(), e.getMaxY());
 91 };

 94 /**
 95  * @param {number} a
 96  * @param {number} b
 97  * @return {number}
 98  */
 99 jsts.index.strtree.STRtree.prototype.avg = function(a, b) {
100   return (a + b) / 2.0;
101 };
