  7 /**
  8  * Computes the buffer of a geometry, for both positive and negative buffer
  9  * distances.
 10  *
 11  * In GIS, the positive buffer of a geometry is defined as
 12  * the Minkowski sum or difference of the geometry
 13  * with a circle of radius equal to the absolute value of the buffer distance.
 14  * In the CAD/CAM world buffers are known as </i>offset curves</i>.
 15  * In morphological analysis they are known as <i>erosion</i> and
 16  * <i>dilation</i>
 17  *
 18  * The buffer operation always returns a polygonal result.
 19  * The negative or zero-distance buffer of lines and points is always an empty
 20  * {@link Polygon}.
 21  *
 22  * Since true buffer curves may contain circular arcs,
 23  * computed buffer polygons can only be approximations to the true geometry.
 24  * The user can control the accuracy of the curve approximation by specifying
 25  * the number of linear segments used to approximate curves.
 26  *
 27  * The <b>end cap style</b> of a linear buffer may be specified. The
 28  * following end cap styles are supported:
 29  * <ul
 30  * <li>{@link #CAP_ROUND} - the usual round end caps
 31  * <li>{@link #CAP_BUTT} - end caps are truncated flat at the line ends
 32  * <li>{@link #CAP_SQUARE} - end caps are squared off at the buffer distance
 33  * beyond the line ends
 34  * </ul>
 35  *
 36  */
 37 
 38 
 39 
 40 /**
 41  * Initializes a buffer computation for the given geometry with the given set
 42  * of parameters.
 43  *
 44  * @param {Geometry}
 45  *          g the geometry to buffer.
 46  * @param {BufferParameters}
 47  *          bufParams the buffer parameters to use.
 48  * @constructor
 49  */
 50 jsts.operation.buffer.BufferOp = function(g, bufParams) {
 51   this.argGeom = g;
 52   this.bufParams = bufParams ? bufParams : new jsts.operation.buffer.BufferParameters();
 53 };

 56 /**
 57  * A number of digits of precision which leaves some computational "headroom"
 58  * for floating point operations.
 59  *
 60  * This value should be less than the decimal precision of double-precision
 61  * values (16).
 62  *
 63  * @type {int}
 64  */
 65 jsts.operation.buffer.BufferOp.MAX_PRECISION_DIGITS = 12;
 66 
 67 
 68 /**
 69  * Compute a scale factor to limit the precision of a given combination of
 70  * Geometry and buffer distance. The scale factor is determined by a
 71  * combination of the number of digits of precision in the (geometry + buffer
 72  * distance), limited by the supplied <code>maxPrecisionDigits</code> value.
 73  *
 74  * @param {Geometry}
 75  *          g the Geometry being buffered.
 76  * @param {double}
 77  *          distance the buffer distance.
 78  * @param {int}
 79  *          maxPrecisionDigits the max # of digits that should be allowed by
 80  *          the precision determined by the computed scale factor.
 81  *
 82  * @return {double} a scale factor for the buffer computation.
 83  */
 84 jsts.operation.buffer.BufferOp.precisionScaleFactor = function(g, distance,
 85     maxPrecisionDigits) {
 86   var env = g.getEnvelopeInternal();
 87   var envSize = Math.max(env.getHeight(), env.getWidth());
 88   var expandByDistance = distance > 0.0 ? distance : 0.0;
 89   var bufEnvSize = envSize + 2 * expandByDistance;
 90 
 91   // the smallest power of 10 greater than the buffer envelope
 92   var bufEnvLog10 = (Math.log(bufEnvSize) / Math.log(10) + 1.0);
 93   var minUnitLog10 = bufEnvLog10 - maxPrecisionDigits;
 94   // scale factor is inverse of min Unit size, so flip sign of exponent
 95   var scaleFactor = Math.pow(10.0, -minUnitLog10);
 96   return scaleFactor;
 97 };

100 /**
101  * Computes the buffer of a geometry for a given buffer distance.
102  *
103  * @param {Geometry}
104  *          g the geometry to buffer.
105  * @param {double}
106  *          distance the buffer distance.
107  * @return {Geometry} the buffer of the input geometry.
108  */
109 jsts.operation.buffer.BufferOp.bufferOp = function(g, distance) {
110   var gBuf = new jsts.operation.buffer.BufferOp(g);
111   var geomBuf = gBuf.getResultGeometry(distance);
112   return geomBuf;
113 };

209 /**
210  * Returns the buffer computed for a geometry for a given buffer distance.
211  *
212  * @param {double}
213  *          dist the buffer distance.
214  * @return {Geometry} the buffer of the input geometry.
215  */
216 jsts.operation.buffer.BufferOp.prototype.getResultGeometry = function(dist) {
217   this.distance = dist;
218   this.computeGeometry();
219   return resultGeometry;
220 };

222 
223 /**
224  * TODO: doc
225  */
226 jsts.operation.buffer.BufferOp.prototype.computeGeometry = function() {
227   this.bufferOriginalPrecision();
228   if (resultGeometry !== null) {
229     return;
230   }
231 
232   var argPM = argGeom.getPrecisionModel();
233   if (argPM.getType() === PrecisionModel.FIXED) {
234     this.bufferFixedPrecision(argPM);
235   }
236   else {
237     this.bufferReducedPrecision();
238   }
239 };

241 
242 /**
243  * TODO: doc
244  */
245 jsts.operation.buffer.BufferOp.prototype.bufferReducedPrecision = function() {
246   var precDigits;
247 
248   // try and compute with decreasing precision
249   for (precDigits = MAX_PRECISION_DIGITS; precDigits >= 0; precDigits--) {
250     try {
251       this.bufferReducedPrecision(precDigits);
252     } catch (/* TopologyException */ex) {
253       saveException = ex;
254       // don't propagate the exception - it will be detected by fact that
255       // resultGeometry is null
256     }
257     if (resultGeometry !== null) {
258       return;
259     }
260   }
261 
262   // tried everything - have to bail
263   throw saveException;
264 };

267 /**
268  * TODO: doc
269  */
270 jsts.operation.buffer.BufferOp.prototype.bufferOriginalPrecision = function() {
271   // use fast noding by default
272   var bufBuilder = new jsts.operation.buffer.BufferBuilder(bufParams);
273   resultGeometry = bufBuilder.buffer(argGeom, distance);
274 };

277 /**
278  * TODO: doc
279  *
280  * @param {int}
281  *          precisionDigits TODO: doc.
282  */
283 jsts.operation.buffer.BufferOp.prototype.bufferReducedPrecision = function(
284     precisionDigits) {
285 
286   var sizeBasedScaleFactor = this.precisionScaleFactor(argGeom, distance,
287       precisionDigits);
288 
289   var fixedPM = new jsts.geom.PrecisionModel(sizeBasedScaleFactor);
290   this.bufferFixedPrecision(fixedPM);
291 };

294 /**
295  * TODO: doc
296  *
297  * @param {PrecisionModel}
298  *          fixedPM TODO: doc.
299  */
300 jsts.operation.buffer.BufferOp.prototype.bufferFixedPrecision = function(
301     fixedPM) {
302   var noder = new ScaledNoder(new MCIndexSnapRounder(new PrecisionModel(1.0)),
303       fixedPM.getScale());
304 
305   var bufBuilder = new jsts.operation.buffer.BufferBuilder(bufParams);
306   bufBuilder.setWorkingPrecisionModel(fixedPM);
307   bufBuilder.setNoder(noder);
308   // this may throw an exception, if robustness errors are encountered
309   resultGeometry = bufBuilder.buffer(argGeom, distance);
310 };
