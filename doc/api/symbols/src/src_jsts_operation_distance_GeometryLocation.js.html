  8 /**
  9  * Represents the location of a point on a Geometry.
 10  * Maintains both the actual point location
 11  * (which may not be exact, if the point is not a vertex)
 12  * as well as information about the component
 13  * and segment index where the point occurs.
 14  * Locations inside area Geometrys will not have an associated segment index,
 15  * so in this case the segment index will have the sentinel value of
 16  * {@link #INSIDE_AREA}.
 17  */
 18 
 19 
 20 
 21 /**
 22  * Constructs a GeometryLocation specifying a point on a geometry, as well as
 23  * the segment that the point is on (or {@link INSIDE_AREA} if the point is not
 24  * on a segment).
 25  *
 26  * @param {Geometry}
 27  *          component the component of the geometry containing the point.
 28  * @param {int}
 29  *          segIndex the segment index of the location, or INSIDE_AREA.
 30  * @param {Coordinate}
 31  *          pt the coordinate of the location.
 32  * @constructor
 33  */
 34 jsts.geom.distance.GeometryLocation = function(component, segIndex, pt) {
 35   this.component = component;
 36   this.segIndex = segIndex;
 37   this.pt = pt;
 38 };
 39 
 40 
 41 /**
 42  * A special value of segmentIndex used for locations inside area geometries.
 43  * These locations are not located on a segment, and thus do not have an
 44  * associated segment index.
 45  *
 46  * @type {int}
 47  */
 48 jsts.geom.distance.GeometryLocation.INSIDE_AREA = -1;
 49 
 50 
 51 /**
 52  * @type {Geometry}
 53  */
 54 jsts.geom.distance.GeometryLocation.prototype.component = null;
 55 
 56 
 57 /**
 58  * @type {int}
 59  */
 60 jsts.geom.distance.GeometryLocation.prototype.segIndex = null;
 61 
 62 
 63 /**
 64  * @type {Coordinate}
 65  */
 66 jsts.geom.distance.GeometryLocation.prototype.pt = null;
 67 
 68 
 69 /**
 70  * @return {Geometry} the geometry component on (or in) which this location
 71  *         occurs.
 72  */
 73 jsts.geom.distance.GeometryLocation.prototype.getGeometryComponent = function() {
 74   return this.component;
 75 };
 76 
 77 
 78 /**
 79  * Returns the segment index for this location. If the location is inside an
 80  * area, the index will have the value {@link INSIDE_AREA};
 81  *
 82  * @return {int} the segment index for the location, or INSIDE_AREA.
 83  */
 84 jsts.geom.GeometryLocation.prototype.getSegmentIndex = function() {
 85   return this.segIndex;
 86 };
 87 
 88 
 89 /**
 90  * @return {Coordinate} the {@link Coordinate} of this location.
 91  */
 92 jsts.geom.distance.GeometryLocation.prototype.getCoordinate = function() {
 93   return this.pt;
 94 };
 95 
 96 
 97 /**
 98  * @return {boolean} whether this location represents a point inside an area
 99  *         geometry.
100  */
101 jsts.geom.distance.GeometryLocation.prototype.isInsideArea = function() {
102   return this.segIndex === jsts.geom.GeometryLocation.INSIDE_AREA;
103 };
