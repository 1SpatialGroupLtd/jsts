  5 /**
  6  * An ordered list of {@link EdgeEndBundle}s around a {@link RelateNode}.
  7  * They are maintained in CCW order (starting with the positive x-axis) around the node
  8  * for efficient lookup and topology building.
  9  *
 10  * @constructor
 11  */
 12 jsts.operation.relate.EdgeEndBundleStar = function() {
 13   jsts.geomgraph.EdgeEndStar.apply(this, arguments);
 14 };
 15 
 16 jsts.operation.relate.EdgeEndBundleStar.prototype = new jsts.geomgraph.EdgeEndStar();
 17 
 18 
 19 /**
 20  * Insert a EdgeEnd in order in the list. If there is an existing EdgeStubBundle
 21  * which is parallel, the EdgeEnd is added to the bundle. Otherwise, a new
 22  * EdgeEndBundle is created to contain the EdgeEnd. <br>
 23  */
 24 jsts.operation.relate.EdgeEndBundleStar.prototype.insert = function(e) {
 25   var eb = this.edgeMap[e];
 26 
 27   if (eb === undefined || !(eb instanceof jsts.operation.relate.EdgeEndBundle)) {
 28     eb = new jsts.operation.relate.EdgeEndBundle(e);
 29     this.insertEdgeEnd(e, eb);
 30   } else {
 31     eb.insert(e);
 32   }
 33 };
 34 
 35 
 36 /**
 37  * Update the IM with the contribution for the EdgeStubs around the node.
 38  */
 39 jsts.operation.relate.EdgeEndBundleStar.prototype.updateIM = function(im) {
 40   var edges = this.getEdges();
 41   for (var i = 0; i < edges.length; i++) {
 42     var esb = edges[i];
 43     esb.updateIM(im);
 44   }
 45 };
