  9 /**
 10  * Computes the topological relationship between two Geometries.
 11  * <p>
 12  * RelateComputer does not need to build a complete graph structure to compute
 13  * the IntersectionMatrix. The relationship between the geometries can be
 14  * computed by simply examining the labelling of edges incident on each node.
 15  * <p>
 16  * RelateComputer does not currently support arbitrary GeometryCollections. This
 17  * is because GeometryCollections can contain overlapping Polygons. In order to
 18  * correct compute relate on overlapping Polygons, they would first need to be
 19  * noded and merged (if not explicitly, at least implicitly).
 20  *
 21  * @constructor
 22  */
 23 jsts.operation.relate.RelateComputer = function(arg) {
 24   this.li = new jsts.algorithm.RobustLineIntersector();
 25   this.ptLocator = new jsts.algorithm.PointLocator();
 26   this.nodes = new jsts.geomgraph.NodeMap(
 27       new jsts.operation.relate.RelateNodeFactory());
 28   this.isolatedEdges = [];
 29 
 30 
 31   this.arg = arg;
 32 };

 90 jsts.operation.relate.RelateComputer.prototype.computeIM = function() {
 91   var im = new jsts.geom.IntersectionMatrix();
 92   // since Geometries are finite and embedded in a 2-D space, the EE element
 93   // must always be 2
 94   im.set(jsts.geom.Location.EXTERIOR, jsts.geom.Location.EXTERIOR, 2);
 95 
 96   // if the Geometries don't overlap there is nothing to do
 97   if (!this.arg[0].getGeometry().getEnvelopeInternal().intersects(
 98       this.arg[1].getGeometry().getEnvelopeInternal())) {
 99     this.computeDisjointIM(im);
100     return im;
101   }
102   this.arg[0].computeSelfNodes(this.li, false);
103   this.arg[1].computeSelfNodes(this.li, false);
104 
105   // compute intersections between edges of the two input geometries
106   var intersector = this.arg[0].computeEdgeIntersections(this.arg[1], this.li, false);
107   // System.out.println("computeIM: # segment intersection tests: " +
108   // intersector.numTests);
109   this.computeIntersectionNodes(0);
110   this.computeIntersectionNodes(1);
111   /**
112    * Copy the labelling for the nodes in the parent Geometries. These override
113    * any labels determined by intersections between the geometries.
114    */
115   this.copyNodesAndLabels(0);
116   this.copyNodesAndLabels(1);
117 
118   // complete the labelling for any nodes which only have a label for a single
119   // geometry
120   this.labelIsolatedNodes();
121 
122   // If a proper intersection was found, we can set a lower bound on the IM.
123   this.computeProperIntersectionIM(intersector, im);
124 
125   /**
126    * Now process improper intersections (eg where one or other of the geometries
127    * has a vertex at the intersection point) We need to compute the edge graph
128    * at all nodes to determine the IM.
129    */
130 
131   // build EdgeEnds for all intersections
132   var eeBuilder = new jsts.operation.relate.EdgeEndBuilder();
133   var ee0 = eeBuilder.computeEdgeEnds(this.arg[0].getEdges());
134   this.insertEdgeEnds(ee0);
135   var ee1 = eeBuilder.computeEdgeEnds(this.arg[1].getEdges());
136   this.insertEdgeEnds(ee1);
137 
138   this.labelNodeEdges();
139 
140   /**
141    * Compute the labeling for isolated components <br>
142    * Isolated components are components that do not touch any other components
143    * in the graph. They can be identified by the fact that they will contain
144    * labels containing ONLY a single element, the one for their parent geometry.
145    * We only need to check components contained in the input graphs, since
146    * isolated components will not have been replaced by new components formed by
147    * intersections.
148    */
149   this.labelIsolatedEdges(0, 1);
150   this.labelIsolatedEdges(1, 0);
151 
152   // update the IM from all components
153   this.updateIM(im);
154   return im;
155 };

158 /**
159  * @private
160  */
161 jsts.operation.relate.RelateComputer.prototype.insertEdgeEnds = function(ee) {
162   for (var i = 0; i < ee.length; i++) {
163     var e = ee[i];
164     this.nodes.add(e);
165   }
166 };

169 /**
170  * @private
171  */
172 jsts.operation.relate.RelateComputer.prototype.computeProperIntersectionIM = function(
173     intersector, im) {
174   // If a proper intersection is found, we can set a lower bound on the IM.
175   var dimA = this.arg[0].getGeometry().getDimension();
176   var dimB = this.arg[1].getGeometry().getDimension();
177   var hasProper = intersector.hasProperIntersection();
178   var hasProperInterior = intersector.hasProperInteriorIntersection();
179 
180   // For Geometry's of dim 0 there can never be proper intersections.
181 
182   /**
183    * If edge segments of Areas properly intersect, the areas must properly
184    * overlap.
185    */
186   if (dimA === 2 && dimB === 2) {
187     if (hasProper)
188       im.setAtLeast('212101212');
189   }
190   /**
191    * If an Line segment properly intersects an edge segment of an Area, it
192    * follows that the Interior of the Line intersects the Boundary of the Area.
193    * If the intersection is a proper <i>interior</i> intersection, then there
194    * is an Interior-Interior intersection too. Note that it does not follow that
195    * the Interior of the Line intersects the Exterior of the Area, since there
196    * may be another Area component which contains the rest of the Line.
197    */
198   else if (dimA === 2 && dimB === 1) {
199     if (hasProper)
200       im.setAtLeast('FFF0FFFF2');
201     if (hasProperInterior)
202       im.setAtLeast('1FFFFF1FF');
203   } else if (dimA === 1 && dimB === 2) {
204     if (hasProper)
205       im.setAtLeast('F0FFFFFF2');
206     if (hasProperInterior)
207       im.setAtLeast('1F1FFFFFF');
208   }
209   /* If edges of LineStrings properly intersect *in an interior point*, all
210           we can deduce is that
211           the interiors intersect.  (We can NOT deduce that the exteriors intersect,
212           since some other segments in the geometries might cover the points in the
213           neighbourhood of the intersection.)
214           It is important that the point be known to be an interior point of
215           both Geometries, since it is possible in a self-intersecting geometry to
216           have a proper intersection on one segment that is also a boundary point of another segment.
217       */
218   else if (dimA === 1 && dimB === 1) {
219     if (hasProperInterior)
220       im.setAtLeast('0FFFFFFFF');
221   }
222 };

225 /**
226  * Copy all nodes from an arg geometry into this graph. The node label in the
227  * arg geometry overrides any previously computed label for that argIndex. (E.g.
228  * a node may be an intersection node with a computed label of BOUNDARY, but in
229  * the original arg Geometry it is actually in the interior due to the Boundary
230  * Determination Rule)
231  *
232  * @private
233  */
234 jsts.operation.relate.RelateComputer.prototype.copyNodesAndLabels = function(
235     argIndex) {
236   var nodes = this.arg[argIndex].getNodes();
237   for (var i = 0; i < nodes.length; i++) {
238     var graphNode = nodes[i];
239     var newNode = this.nodes.addNode(graphNode.getCoordinate());
240     newNode.setLabel(argIndex, graphNode.getLabel().getLocation(argIndex));
241   }
242 };

245 /**
246  * Insert nodes for all intersections on the edges of a Geometry. Label the
247  * created nodes the same as the edge label if they do not already have a label.
248  * This allows nodes created by either self-intersections or mutual
249  * intersections to be labelled. Endpoint nodes will already be labelled from
250  * when they were inserted.
251  *
252  * @private
253  */
254 jsts.operation.relate.RelateComputer.prototype.computeIntersectionNodes = function(
255     argIndex) {
256   var edges = this.arg[argIndex].getEdges();
257   for (var i = 0; i < edges.length; i++) {
258     var e = edges[i];
259     var eLoc = e.getLabel().getLocation(argIndex);
260     var eis = e.eiList.getSortedIntersections();
261     for (var j = 0; j < eis.length; j++) {
262       var ei = eis[j];
263       var n = this.nodes.addNode(ei.coord);
264       if (eLoc === jsts.geom.Location.BOUNDARY)
265         n.setLabelBoundary(argIndex);
266       else {
267         if (n.getLabel().isNull(argIndex))
268           n.setLabel(argIndex, jsts.geom.Location.INTERIOR);
269       }
270     }
271   }
272 };

275 /**
276  * For all intersections on the edges of a Geometry, label the corresponding
277  * node IF it doesn't already have a label. This allows nodes created by either
278  * self-intersections or mutual intersections to be labelled. Endpoint nodes
279  * will already be labelled from when they were inserted.
280  *
281  * @private
282  */
283 jsts.operation.relate.RelateComputer.prototype.labelIntersectionNodes = function(
284     argIndex) {
285   var edges = this.arg[argIndex].getEdges();
286   for (var i = 0; i < edges.length; i++) {
287     var e = edges[i];
288     var eLoc = e.getLabel().getLocation(argIndex);
289     var eis = e.eiList.getSortedIntersections();
290     for (var j = 0; j < eis.length; j++) {
291       var ei = eis[j];
292       var n = this.nodes.find(ei.coord);
293       if (n.getLabel().isNull(argIndex)) {
294         if (eLoc === jsts.geom.Location.BOUNDARY)
295           n.setLabelBoundary(argIndex);
296         else
297           n.setLabel(argIndex, jsts.geom.Location.INTERIOR);
298       }
299     }
300   }
301 };

304 /**
305  * If the Geometries are disjoint, we need to enter their dimension and boundary
306  * dimension in the Ext rows in the IM
307  *
308  * @private
309  */
310 jsts.operation.relate.RelateComputer.prototype.computeDisjointIM = function(im) {
311   var ga = this.arg[0].getGeometry();
312   if (!ga.isEmpty()) {
313     im.set(jsts.geom.Location.INTERIOR, jsts.geom.Location.EXTERIOR, ga
314         .getDimension());
315     im.set(jsts.geom.Location.BOUNDARY, jsts.geom.Location.EXTERIOR, ga
316         .getBoundaryDimension());
317   }
318   var gb = this.arg[1].getGeometry();
319   if (!gb.isEmpty()) {
320     im.set(jsts.geom.Location.EXTERIOR, jsts.geom.Location.INTERIOR, gb
321         .getDimension());
322     im.set(jsts.geom.Location.EXTERIOR, jsts.geom.Location.BOUNDARY, gb
323         .getBoundaryDimension());
324   }
325 };

328 /**
329  * @private
330  */
331 jsts.operation.relate.RelateComputer.prototype.labelNodeEdges = function() {
332   var nodes = this.nodes.values();
333   for (var i = 0; i < nodes.length; i++) {
334     var node = nodes[i];
335     node.getEdges().computeLabelling(this.arg);
336   }
337 };

340 /**
341  * update the IM with the sum of the IMs for each component
342  *
343  * @private
344  */
345 jsts.operation.relate.RelateComputer.prototype.updateIM = function(im) {
346   for (var ei = 0; ei < this.isolatedEdges.length; ei++) {
347     var e = this.isolatedEdges[ei];
348     e.updateIM(im);
349   }
350   var nodes = this.nodes.values();
351   for (var i = 0; i < nodes.length; i++) {
352     var node = nodes[i];
353     node.updateIM(im);
354     node.updateIMFromEdges(im);
355   }
356 };

359 /**
360  * Processes isolated edges by computing their labelling and adding them to the
361  * isolated edges list. Isolated edges are guaranteed not to touch the boundary
362  * of the target (since if they did, they would have caused an intersection to
363  * be computed and hence would not be isolated)
364  *
365  * @private
366  */
367 jsts.operation.relate.RelateComputer.prototype.labelIsolatedEdges = function(
368     thisIndex, targetIndex) {
369   var edges = this.arg[thisIndex].edges;
370   for (var i = 0; i < edges.length; i++) {
371     var e = edges[i];
372     if (e.isIsolated()) {
373       this.labelIsolatedEdge(e, targetIndex, this.arg[targetIndex].getGeometry());
374       this.isolatedEdges.push(e);
375     }
376   }
377 };

380 /**
381  * Label an isolated edge of a graph with its relationship to the target
382  * geometry. If the target has dim 2 or 1, the edge can either be in the
383  * interior or the exterior. If the target has dim 0, the edge must be in the
384  * exterior
385  *
386  * @private
387  */
388 jsts.operation.relate.RelateComputer.prototype.labelIsolatedEdge = function(e,
389     targetIndex, target) {
390   // this won't work for GeometryCollections with both dim 2 and 1 geoms
391   if (target.getDimension() > 0) {
392     // since edge is not in boundary, may not need the full generality of
393     // PointLocator?
394     // Possibly should use ptInArea locator instead? We probably know here
395     // that the edge does not touch the bdy of the target Geometry
396     var loc = this.ptLocator.locate(e.getCoordinate(), target);
397     e.getLabel().setAllLocations(targetIndex, loc);
398   } else {
399     e.getLabel().setAllLocations(targetIndex, jsts.geom.Location.EXTERIOR);
400   }
401 };

404 /**
405  * Isolated nodes are nodes whose labels are incomplete (e.g. the location for
406  * one Geometry is null). This is the case because nodes in one graph which
407  * don't intersect nodes in the other are not completely labelled by the initial
408  * process of adding nodes to the nodeList. To complete the labelling we need to
409  * check for nodes that lie in the interior of edges, and in the interior of
410  * areas.
411  *
412  * @private
413  */
414 jsts.operation.relate.RelateComputer.prototype.labelIsolatedNodes = function() {
415   var nodes = this.nodes.values();
416   for (var i = 0; i < nodes.length; i++) {
417     var n = nodes[i];
418     var label = n.getLabel();
419     // isolated nodes should always have at least one geometry in their label
420     // Assert.isTrue(label.getGeometryCount() > 0, 'node with empty label
421     // found');
422     if (n.isIsolated()) {
423       if (label.isNull(0))
424         this.labelIsolatedNode(n, 0);
425       else
426         this.labelIsolatedNode(n, 1);
427     }
428   }
429 };

432 /**
433  * Label an isolated node with its relationship to the target geometry.
434  */
435 jsts.operation.relate.RelateComputer.prototype.labelIsolatedNode = function(n,
436     targetIndex) {
437   var loc = this.ptLocator.locate(n.getCoordinate(), this.arg[targetIndex]
438       .getGeometry());
439   n.getLabel().setAllLocations(targetIndex, loc);
440 };
