  9 /**
 10  * Implements the simple graph of Nodes and EdgeEnd which is all that is
 11  * required to determine topological relationships between Geometries. Also
 12  * supports building a topological graph of a single Geometry, to allow
 13  * verification of valid topology.
 14  * <p>
 15  * It is <b>not</b> necessary to create a fully linked PlanarGraph to determine
 16  * relationships, since it is sufficient to know how the Geometries interact
 17  * locally around the nodes. In fact, this is not even feasible, since it is not
 18  * possible to compute exact intersection points, and hence the topology around
 19  * those nodes cannot be computed robustly. The only Nodes that are created are
 20  * for improper intersections; that is, nodes which occur at existing vertices
 21  * of the Geometries. Proper intersections (e.g. ones which occur between the
 22  * interior of line segments) have their topology determined implicitly, without
 23  * creating a Node object to represent them.
 24  *
 25  * @constructor
 26  */
 27 jsts.operation.relate.RelateNodeGraph = function() {
 28   this.nodes = new jsts.geomgraph.NodeMap(
 29       new jsts.operation.relate.RelateNodeFactory());
 30 };
 31 
 32 
 33 /**
 34  * @private
 35  */
 36 jsts.operation.relate.RelateNodeGraph.prototype.nodes = null;
 37 
 38 
 39 jsts.operation.relate.RelateNodeGraph.prototype.build = function(geomGraph) {
 40   // compute nodes for intersections between previously noded edges
 41   this.computeIntersectionNodes(geomGraph, 0);
 42   /**
 43    * Copy the labelling for the nodes in the parent Geometry. These override any
 44    * labels determined by intersections.
 45    */
 46   this.copyNodesAndLabels(geomGraph, 0);
 47 
 48   /**
 49    * Build EdgeEnds for all intersections.
 50    */
 51   var eeBuilder = new EdgeEndBuilder();
 52   var eeList = eeBuilder.computeEdgeEnds(geomGraph.edges);
 53   this.insertEdgeEnds(eeList);
 54 
 55   // Debug.println("==== NodeList ===");
 56   // Debug.print(nodes);
 57 };
 58 
 59 
 60 /**
 61  * Insert nodes for all intersections on the edges of a Geometry. Label the
 62  * created nodes the same as the edge label if they do not already have a label.
 63  * This allows nodes created by either self-intersections or mutual
 64  * intersections to be labelled. Endpoint nodes will already be labelled from
 65  * when they were inserted.
 66  * <p>
 67  * Precondition: edge intersections have been computed.
 68  */
 69 jsts.operation.relate.RelateNodeGraph.prototype.computeIntersectionNodes = function(
 70     geomGraph, argIndex) {
 71   var edges = geomGraph.edges;
 72   for (var i = 0; i < edges.length; i++) {
 73     var e = edges[i];
 74     var eLoc = e.getLabel().getLocation(argIndex);
 75     for (var j = 0; j < e.eiList.length; j++) {
 76       var ei = e.eiList[j];
 77       RelateNode;
 78       n = this.nodes.addNode(ei.coord);
 79       if (eLoc === Location.BOUNDARY)
 80         n.setLabelBoundary(argIndex);
 81       else {
 82         if (n.getLabel().isNull(argIndex))
 83           n.setLabel(argIndex, Location.INTERIOR);
 84       }
 85       // Debug.println(n);
 86     }
 87   }
 88 };
 89 
 90 
 91 /**
 92  * Copy all nodes from an arg geometry into this graph. The node label in the
 93  * arg geometry overrides any previously computed label for that argIndex. (E.g.
 94  * a node may be an intersection node with a computed label of BOUNDARY, but in
 95  * the original arg Geometry it is actually in the interior due to the Boundary
 96  * Determination Rule)
 97  */
 98 jsts.operation.relate.RelateNodeGraph.prototype.copyNodesAndLabels = function(
 99     geomGraph, argIndex) {
100   var nodes = geomGraph.nodes;
101   for (var i = 0; i < nodes.length; i++) {
102     var graphNode = nodes[i];
103     var newNode = nodes.addNode(graphNode.getCoordinate());
104     newNode.setLabel(argIndex, graphNode.getLabel().getLocation(argIndex));
105     // node.print(System.out);
106   }
107 };
108 
109 jsts.operation.relate.RelateNodeGraph.prototype.insertEdgeEnds = function(ee) {
110   for (var i = 0; i < ee.length; i++) {
111     var e = ee[i];
112     this.nodes.add(e);
113   }
114 };
