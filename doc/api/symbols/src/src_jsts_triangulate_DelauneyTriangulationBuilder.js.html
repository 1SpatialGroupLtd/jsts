  8 /**
  9  * A utility class which creates Delaunay Trianglulations
 10  * from collections of points and extract the resulting
 11  * triangulation edges or triangles as geometries.
 12  *
 13  */
 14 jsts.triangulate.DelauneyTriangulationBuilder = OpenLayers.Class();
 15 
 16 
 17 /**
 18  * Extracts the unique {@link Coordinate}s from the given {@link Geometry}.
 19  * @param {jsts.geom.Geometry}
 20  *        geom the geometry to extract from.
 21  * @return {Array{jsts.geom.Coordinate}}
 22  *         An array of the unique Coordinates
 23  */
 24 jsts.triangulate.DelauneyTriangulationBuilder.extractUniqueCoordinates =
 25     function(geom) {
 26   if (geom === undefined || geom === null) {
 27     return new jsts.geom.CoordinateList();
 28   }
 29 
 30   var coords = geom.getCoordinates();
 31   return jsts.triangulate.DelauneyTriangulationBuilder.unique(coords);
 32 };

 34 
 35 /**
 36  * Removes any duplicates in the passed array.
 37  *
 38  * @param {Array{jsts.geom.Coordinate}}
 39  *        coords The input coordinates
 40  * @return {Array{jsts.geom.Coordinate}}
 41  *         An array stripped out of any duplicates
 42  */
 43 jsts.triangulate.DelauneyTriangulationBuilder.unique =
 44     function(coords) {
 45   //Sort the coordinates by their compareTo-function
 46   coords.sort(function(a, b) {
 47     return a.compareTo(b);
 48   });
 49 
 50   coordList = new jsts.geom.CoordinateList(coords, false);
 51   return coordList;
 52 };

 55 /**
 56  * Converts an array of coordinates to an array of vertexes
 57  *
 58  * @param {Array{jsts.geom.Coordinate}}
 59  *        coords the input coordinates
 60  * @return {Array{jsts.triangulate.quadedge.Vertex}}
 61  *         The created vertexes
 62  */
 63 jsts.triangulate.DelauneyTriangulationBuilder.toVertices =
 64     function(coords) {
 65   var verts = new Array(coords.length), i = 0, il = coords.length, coord;
 66 
 67   for (i; i < il; i++) {
 68     coord = coords[i];
 69     verts[i] = new jsts.triangulate.quadedge.Vertex(coord);
 70   }
 71 
 72   return verts;
 73 };

 76 /**
 77  * Computes the {jsts.geom.Envelope} of an array of {jsts.geom.Coordinate}s
 78  *
 79  * @param   {Array{jsts.geom.Coordinate}}
 80  *          coords the input coordinates
 81  * @return  {jsts.geom.Envelope}
 82  *          The created envelope.
 83  *
 84  */
 85 jsts.triangulate.DelauneyTriangulationBuilder.envelope =
 86     function(coords) {
 87   var env = new jsts.geom.Envelope(), i = 0, il = coords.length;
 88 
 89   for (i; i < il; i++) {
 90     env.expandToInclude(coords[i]);
 91   }
 92 
 93   return env;
 94 };

 97 /**
 98  * Initializes a new DelauneyTriangulationBuilder
 99  */
100 jsts.triangulate.DelauneyTriangulationBuilder.prototype.initialize =
101     function() {
102   this.siteCoords = null;
103   this.tolerance = 0.0;
104   this.subdiv = null;
105 };

108 /**
109  * Sets the sites which will be triangulated.
110  * Calls the correct setSites* function after argument-checking
111  */
112 jsts.triangulate.DelauneyTriangulationBuilder.prototype.setSites = function() {
113   var arg = arguments[0];
114 
115   if (arg instanceof jsts.geom.Geometry ||
116       arg instanceof jsts.geom.Coordinate ||
117       arg instanceof jsts.geom.Point || arg instanceof jsts.geom.MultiPoint ||
118       arg instanceof jsts.geom.LineString ||
119       arg instanceof jsts.geom.MultiLineString ||
120       arg instanceof jsts.geom.LinearRing || arg instanceof jsts.geom.Polygon ||
121       arg instanceof jsts.geom.MultiPolygon) {
122     this.setSitesFromGeometry(arg);
123   }else {
124     this.setSitesFromCollection(arg);
125   }
126 };
127 
128 
129 /**
130  * Sets the sites (point or vertices) which will be triangulated.
131  * All vertices of the given geometry will be used as sites.
132  *
133  * @param {jsts.geom.Geometry}
134  *        geom the geometry from which the sites will be extracted.
135  */
136 jsts.triangulate.DelauneyTriangulationBuilder.prototype.setSitesFromGeometry =
137     function(geom) {
138   // remove any duplicate points (they will cause the triangulation to fail)
139   this.siteCoords = jsts.triangulate.DelauneyTriangulationBuilder.extractUniqueCoordinates(geom);
140 };
141 
142 
143 /**
144  * Sets the sites (point or vertices) which will be triangulated
145  * from a collection of {@link Coordinate}s.
146  *
147  * @param {Array{Coordinates}}
148  *        coords a collection of Coordinates.
149  */
150 jsts.triangulate.DelauneyTriangulationBuilder.prototype
151   .setSitesFromCollection = function(coords) {
152       // remove any duplicate points (they will cause the triangulation to fail)
153       this.siteCoords = jsts.triangulate.DelauneyTriangulationBuilder.extractUniqueCoordinates.unique(coords);
154     };

157 /**
158  * Sets the snapping tolerance which will be used
159  * to improved the robustness of the triangulation computation.
160  * A tolerance of 0.0 specifies that no snapping will take place.
161  *
162  * @param {Number}
163  *        tolerance the tolerance distance to use.
164  */
165 jsts.triangulate.DelauneyTriangulationBuilder.prototype.setTolerance =
166     function(tolerance) {
167   this.tolerance = tolerance;
168 };

171 /**
172  * Creates the delauney-triangulation.
173  *
174  */
175 jsts.triangulate.DelauneyTriangulationBuilder.prototype.create = function() {
176   if (this.subdiv === null) {
177     var siteEnv, vertices, triangulator;
178 
179     siteEnv = jsts.triangulate.DelauneyTriangulationBuilder.envelope(this.siteCoords);
180     vertices = jsts.triangulate.DelauneyTriangulationBuilder.toVertices(this.siteCoords);
181     this.subdiv = new jsts.triangulate.quadedge.
182         QuadEdgeSubdivision(siteEnv, this.tolerance);
183     triangulator = new jsts.triangulate.
184         IncrementalDelauneyTriangulator(this.subdiv);
185     triangulator.insertSites(vertices);
186   }
187 };

190 /**
191  * Gets the {jsts.triangulate.quadedge.QuadEdgeSubdivision} which models
192  * the computed triangulation.
193  *
194  * @return {jsts.triangulate.quadedge.QuadEdgeSubdivision} containing the
195  * triangulation.
196  */
197 jsts.triangulate.DelauneyTriangulationBuilder.prototype.getSubdivision =
198     function() {
199   this.create();
200   return this.subdiv;
201 };

204 /**
205  * Gets the edges of the computed triangulation as a {@link MultiLineString}.
206  *
207  * @param {jsts.geom.GeometryFactory}
208  *        geomFact the geometry factory to use to create the output.
209  * @return {jsts.geom.Geometry}
210  *         the edges of the triangulation.
211  */
212 jsts.triangulate.DelauneyTriangulationBuilder.prototype.
213     getEdges = function(geomFact) {
214   this.create();
215   return this.subdiv.getEdges(geomFact);
216 };

222  *
223  * @param {jsts.geom.GeometryFactory}
224  *        geomFact the geometry factory to use to create the output.

228 jsts.triangulate.DelauneyTriangulationBuilder.prototype.getTriangles =
